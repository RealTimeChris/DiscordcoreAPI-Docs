45 /// For creating a Guild. \brief For creating a Guild.
46 struct DiscordCoreAPI_Dll CreateGuildData {
47 AfkTimeOutDurations afkTimeout{ AfkTimeOutDurations::Shortest };///< Afk timeout in seconds.
48 DefaultMessageNotificationLevel defaultMessageNotifications{};///< Default message notification level.
49 ExplicitContentFilterLevel explicitContentFilter{};///< Explicit content filter level.
50 std::vector<ChannelData> channels{};///< Array of partial Channel objects.
51 int32_t systemChannelFlags{ 0 };///< System Channel flags.
52 int32_t verificationLevel{ 0 };///< Verification level.
53 std::vector<Role> roles{};///< Array of Role objects.
54 Snowflake systemChannelId{};///< The id of the Channel where Guild notices such as welcome messages and boost events are posted.
55 std::vector<uint8_t> icon{};///< base64 128x128 image for the Guild icon.
56 Snowflake afkChannelId{};///< Id for afk Channel.
57 std::string region{};///< The region that the servers are in.
58 std::string name{};///< The name of the new Guild.
59
60 operator std::string() {
61 nlohmann::json data{};
62 data["default_message_notifications"] = this->defaultMessageNotifications;
63 data["explicit_content_filter"] = this->explicitContentFilter;
64 data["system_channel_flags"] = this->systemChannelFlags;
65 data["verification_level"] = this->verificationLevel;
66 data["afk_timeout"] = this->afkTimeout;
67 data["region"] = this->region;
68 data["name"] = this->name;
69 data["icon"] = this->icon;
70 for (auto& value: this->channels) {
71 nlohmann::json newData{};
72 newData["parent_id"] = std::to_string(value.parentId);
73 newData["name"] = std::string{ value.name };
74 newData["id"] = value.id;
75 newData["type"] = value.type;
76 data["channels"].push_back(newData);
77 }
78 for (auto& value: this->roles) {
79 nlohmann::json newData{};
80 newData["permissions"] = value.permissions.getCurrentPermissionString();
81 newData["tags"]["premium_subscriber"] = value.tags.premiumSubscriber;
82 newData["tags"]["integration_id"] = value.tags.integrationId;
83 newData["mentionable"] = DiscordCoreAPI::getBool<int8_t, DiscordCoreAPI::RoleFlags>(value.flags, DiscordCoreAPI::RoleFlags::Mentionable);
84 newData["tags"]["bot_id"] = value.tags.botId;
85 newData["managed"] = DiscordCoreAPI::getBool<int8_t, DiscordCoreAPI::RoleFlags>(value.flags, DiscordCoreAPI::RoleFlags::Managed);
86 newData["position"] = value.position;
87 newData["hoist"] = DiscordCoreAPI::getBool<int8_t, DiscordCoreAPI::RoleFlags>(value.flags, DiscordCoreAPI::RoleFlags::Hoist);
88 newData["color"] = value.color;
89 newData["name"] = value.name;
90 data["roles"].push_back(newData);
91 }
92 if (this->systemChannelId != 0) {
93 data["system_channel_id"] = this->systemChannelId;
94 }
95 if (this->afkChannelId != 0) {
96 data["afk_channel_id"] = this->afkChannelId;
97 }
98 return data.dump();
99 }
100 };

407 /// For modifying the properties of a chosen Guild. \brief For modifying the properties of a chosen Guild.
