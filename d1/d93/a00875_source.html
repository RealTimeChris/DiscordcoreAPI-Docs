195 CoRoutine<std::vector<Role>> Roles::modifyGuildRolePositionsAsync(ModifyGuildRolePositionsData dataPackage) {
196 DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Patch_Guild_Role_Positions };
197 co_await NewThreadAwaitable<std::vector<Role>>();
198 std::vector<Role> currentRoles = Roles::getGuildRolesAsync({ .guildId = dataPackage.guildId }).get();
199 Role newRole = Roles::getCachedRoleAsync({ .roleId = dataPackage.roleId }).get();
200 for (auto& value: currentRoles) {
201 if (value.id == newRole.id) {
202 continue;
203 }
204 RolePositionData newData;
205 if (newRole.position < dataPackage.newPosition) {
206 if (value.position <= dataPackage.newPosition && value.position > newRole.position) {
207 newData.roleId = value.id;
208 newData.rolePosition = value.position - 1;
209 dataPackage.rolePositions.push_back(newData);
210 }
211 } else {
212 if (value.position >= dataPackage.newPosition && value.position < newRole.position) {
213 newData.roleId = value.id;
214 newData.rolePosition = value.position + 1;
215 dataPackage.rolePositions.push_back(newData);
216 }
217 }
218 }
219 RolePositionData newDataPos;
220 newDataPos.roleId = dataPackage.roleId;
221 newDataPos.rolePosition = dataPackage.newPosition;
222 dataPackage.rolePositions.push_back(newDataPos);
223 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Patch;
224 workload.relativePath = "/guilds/" + std::to_string(dataPackage.guildId) + "/roles";
225 workload.content = dataPackage;
226 workload.callStack = "Roles::modifyGuildRolePositionsAsync()";
227 if (dataPackage.reason != "") {
228 workload.headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
229 }
230 co_return Roles::httpsClient->submitWorkloadAndGetResult<RoleVector>(workload);

288 CoRoutine<RoleData> Roles::getCachedRoleAsync(GetRoleData dataPackage) {
289 co_await NewThreadAwaitable<RoleData>();
290 std::shared_lock theLock{ Roles::theMutex };
291 if (!Roles::cache->contains(dataPackage.roleId)) {
292 theLock.unlock();
293 auto theRole = Roles::getRoleAsync(dataPackage).get();
294 Roles::insertRole(theRole);
295 co_return theRole;
296 } else {
297 co_return *(*Roles::cache)[dataPackage.roleId];
298 }

301 void Roles::insertRole(RoleData role) {
302 std::unique_lock theLock{ Roles::theMutex };
303 if (role.id == 0) {
304 return;
305 }
306 auto newCache = std::make_unique<std::unordered_map<Snowflake, std::unique_ptr<RoleData>>>();
307 for (auto& [key, value]: *Roles::cache) {
308 (*newCache)[key] = std::move(value);
309 }
310 if (Roles::configManager->doWeCacheRoles()) {
311 (*newCache)[role.id] = std::make_unique<RoleData>(role);
312 }
313 Roles::cache.reset(nullptr);
314 Roles::cache = std::move(newCache);
315 }

319 Roles::cache->erase(roleId);
