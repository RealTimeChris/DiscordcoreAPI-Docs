526 MoveThroughMessagePagesData moveThroughMessagePages(const std::string& userID, InputEventData originalEvent, uint32_t currentPageIndex,
527 const std::vector<EmbedData>& messageEmbeds, bool deleteAfter, uint32_t waitForMaxMs, bool returnResult) {
528 MoveThroughMessagePagesData returnData{};
529 uint32_t newCurrentPageIndex = currentPageIndex;
530 std::unique_ptr<RespondToInputEventData> dataPackage{ std::make_unique<RespondToInputEventData>(originalEvent) };
531 if (messageEmbeds.size() > 0) {
532 dataPackage->addMessageEmbed(messageEmbeds[currentPageIndex]);
533 }
534 if (returnResult) {
535 dataPackage->addButton(false, "select", "Select", ButtonStyle::Success, "✅");
536 }
537 dataPackage->addButton(false, "backwards", "Prev Page", ButtonStyle::Primary, "◀️");
538 dataPackage->addButton(false, "forwards", "Next Page", ButtonStyle::Primary, "▶️");
539 dataPackage->addButton(false, "exit", "Exit", ButtonStyle::Danger, "❌");
540 dataPackage->setResponseType(InputEventResponseType::Edit_Interaction_Response);
541 originalEvent = InputEvents::respondToInputEventAsync(*dataPackage).get();
542 while (true) {
543 std::unique_ptr<ButtonCollector> button{ std::make_unique<ButtonCollector>(originalEvent) };
544
545 std::vector<ButtonResponseData> buttonIntData{ button->collectButtonData(false, waitForMaxMs, 1, stoull(userID)).get() };
546
547 if (buttonIntData.size() == 0 || buttonIntData.at(0).buttonId == "empty" || buttonIntData.at(0).buttonId == "exit") {
548 std::unique_ptr<RespondToInputEventData> dataPackage02{ std::make_unique<RespondToInputEventData>(originalEvent) };
549 if (buttonIntData.at(0).buttonId == "empty") {
550 *dataPackage02 = originalEvent;
551 } else {
552 std::unique_ptr<InteractionData> interactionData = std::make_unique<InteractionData>(static_cast<InteractionData>(buttonIntData.at(0)));
553 *dataPackage02 = RespondToInputEventData{ *interactionData };
554 }
555
556 dataPackage02->addMessageEmbed(messageEmbeds[newCurrentPageIndex]);
557 for (auto& value: originalEvent.getComponents()) {
558 for (auto& value02: value.components) {
559 value02.disabled = true;
560 }
561 dataPackage02->addComponentRow(value);
562 }
563 if (deleteAfter == true) {
564 InputEventData dataPackage03{ originalEvent };
565 InputEvents::deleteInputEventResponseAsync(dataPackage03);
566 } else {
567 dataPackage02->setResponseType(InputEventResponseType::Edit_Interaction_Response);
568 InputEvents::respondToInputEventAsync(*dataPackage02).get();
569 }
570 MoveThroughMessagePagesData dataPackage03{};
571 dataPackage03.inputEventData = originalEvent;
