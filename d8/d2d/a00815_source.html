548 std::string Permissions::computeOverwrites(const std::string& basePermissions, const GuildMember& guildMember, ChannelData& channel) {
549 if ((stoll(basePermissions) & static_cast<int64_t>(Permission::Administrator)) == static_cast<int64_t>(Permission::Administrator)) {
550 return Permissions::getAllPermissions();
551 }
552
553 int64_t permissions = stoll(basePermissions);
554 if (channel.permissionOverwrites.contains(guildMember.guildId)) {
555 OverWriteData overWritesEveryone = channel.permissionOverwrites[guildMember.guildId];
556 permissions &= ~stoll(static_cast<std::string>(static_cast<StringWrapper>(overWritesEveryone.deny)));
557 permissions |= stoll(static_cast<std::string>(static_cast<StringWrapper>(overWritesEveryone.allow)));
558 }
559
560 std::vector<RoleData> guildMemberRoles{};
561 for (auto& value: guildMember.roles) {
562 guildMemberRoles.push_back(Roles::getCachedRoleAsync({ .guildId = guildMember.guildId, .roleId = value }).get());
563 }
564 int64_t allow{ 0 };
565 int64_t deny{ 0 };
566 for (auto& value: guildMemberRoles) {
567 if (channel.permissionOverwrites.contains(value.id)) {
568 OverWriteData currentChannelOverwrites = channel.permissionOverwrites[value.id];
569 allow |= stoll(static_cast<std::string>(static_cast<StringWrapper>(currentChannelOverwrites.allow)));
570 deny |= stoll(static_cast<std::string>(static_cast<StringWrapper>(currentChannelOverwrites.deny)));
571 }
572 }
573 permissions &= ~deny;
574 permissions |= allow;
575 if (channel.permissionOverwrites.contains(guildMember.id)) {
576 OverWriteData currentOverWrites = channel.permissionOverwrites[guildMember.id];
577 permissions &= ~stoll(static_cast<std::string>(static_cast<StringWrapper>(currentOverWrites.deny)));
578 permissions |= stoll(static_cast<std::string>(static_cast<StringWrapper>(currentOverWrites.allow)));
579 }
580 return std::to_string(permissions);
581 }

583 std::string Permissions::computePermissions(const GuildMember& guildMember, ChannelData& channel) {
584 std::string permissions = Permissions::computeBasePermissions(guildMember);
585 permissions = Permissions::computeOverwrites(permissions, guildMember, channel);
586 return permissions;
587 }

589 std::string Permissions::computeBasePermissions(const GuildMember& guildMember) {
590 const GuildData guild = Guilds::getCachedGuildAsync({ .guildId = guildMember.guildId }).get();
591 if (guild.ownerId == guildMember.id) {
592 return Permissions::getAllPermissions();
593 }
594 std::vector<RoleData> guildRoles{};
595 for (auto& value: guild.roles) {
596 guildRoles.push_back(Roles::getCachedRoleAsync({ .guildId = guild.id, .roleId = value }).get());
597 }
598 RoleData roleEveryone{};
599 for (auto& value: guildRoles) {
600 if (value.id == guild.id) {
601 roleEveryone = value;
602 }
603 }
604 int64_t permissions{};
605 if (roleEveryone.permissions != "") {
606 permissions = stoll(static_cast<std::string>(static_cast<StringWrapper>(roleEveryone.permissions)));
607 }
608 GetGuildMemberRolesData getRolesData{};
609 getRolesData.guildMember = guildMember;
610 getRolesData.guildId = guildMember.guildId;
611 std::vector<RoleData> guildMemberRoles{};
612 for (auto& value: guildMember.roles) {
613 guildMemberRoles.push_back(Roles::getCachedRoleAsync({ .guildId = guild.id, .roleId = value }).get());
614 }
615 for (auto& value: guildMemberRoles) {
616 permissions |= stoll(static_cast<std::string>(static_cast<StringWrapper>(value.permissions)));
617 }
618
619 if ((permissions & static_cast<int64_t>(Permission::Administrator)) == static_cast<int64_t>(Permission::Administrator)) {
620 return Permissions::getAllPermissions();
621 }
622
623 return std::to_string(permissions);
624 }

664 std::string constructMultiPartData(nlohmann::json theData, const std::vector<File>& files) {
665 const std::string boundary("boundary25");
666 const std::string partStart("--" + boundary + "\r\nContent-Type: application/octet-stream\r\nContent-Disposition: form-data; ");
667
668 std::string content("--" + boundary);
669
670 content += "\r\nContent-Type: application/json\r\nContent-Disposition: form-data; "
671 "name=\"payload_json\"\r\n\r\n";
672 content += theData.dump() + "\r\n";
673 if (files.size() == 1) {
674 content += partStart + "name=\"file\"; filename=\"" + files[0].fileName + "\"" + "\r\n\r\n";
675 content += files[0].data;
676 } else {
677 for (uint8_t x = 0; x < files.size(); x++) {
678 content += partStart + "name=\"files[" + std::to_string(x) + "]\"; filename=\"" + files[x].fileName + "\"\r\n\r\n";
679 content += files[x].data;
680 content += "\r\n";
681 }
682 }
683 content += "\r\n--" + boundary + "--";
684 return content;
685 }

687 std::string convertToLowerCase(const std::string& stringToConvert) {
688 std::string newString;
689 for (auto& value: stringToConvert) {
690 if (isupper(static_cast<unsigned char>(value))) {
691 newString += static_cast<char>(tolower(static_cast<unsigned char>(value)));
692 } else {
693 newString += value;
694 }
695 }
696 return newString;
697 }
