475 CoRoutine<void> EventHandler::onChannelCreation(OnChannelCreationData dataPackage) {
476 co_await NewThreadAwaitable<void>();
477 if (EventHandler::configManager->doWeCacheChannels()) {
478 Channels::insertChannel(dataPackage.channel);
479 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.channel.guildId }).get();
480 guild.channels.push_back(dataPackage.channel.id);
481 Guilds::insertGuild(guild);
482 }
483 co_return;
484 }
485
486 CoRoutine<void> EventHandler::onChannelUpdate(OnChannelUpdateData dataPackage) {
487 co_await NewThreadAwaitable<void>();
488 if (EventHandler::configManager->doWeCacheChannels()) {
489 Channels::insertChannel(dataPackage.channelNew);
490 }
491 co_return;
492 }
493
494 CoRoutine<void> EventHandler::onChannelDeletion(OnChannelDeletionData dataPackage) {
495 co_await NewThreadAwaitable<void>();
496 if (EventHandler::configManager->doWeCacheChannels()) {
497 Channels::removeChannel(dataPackage.channel.id);
498 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.channel.guildId }).get();
499 for (uint64_t x = 0; x < guild.channels.size(); x++) {
500 if (guild.channels[x] == dataPackage.channel.id) {
501 guild.channels.erase(guild.channels.begin() + x);
502 }
503 }
504 Guilds::insertGuild(guild);
505 }
506 co_return;
507 }

525 CoRoutine<void> EventHandler::onGuildDeletion(OnGuildDeletionData dataPackage) {
526 co_await NewThreadAwaitable<void>();
527 if (EventHandler::configManager->doWeCacheGuilds()) {
528 for (auto& value: dataPackage.guild.members) {
529 GuildMember guildMember = GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = value, .guildId = dataPackage.guild.id }).get();
530 GuildMembers::removeGuildMember(guildMember);
531 }
532 for (auto& value: dataPackage.guild.channels) {
533 Channels::removeChannel(value);
534 }
535 for (auto& value: dataPackage.guild.roles) {
536 Roles::removeRole(value);
537 }
538 Guilds::removeGuild(dataPackage.guild.id);
539 }
540 co_return;
541 }

543 CoRoutine<void> EventHandler::onGuildMemberAdd(OnGuildMemberAddData dataPackage) {
544 co_await NewThreadAwaitable<void>();
545 if (EventHandler::configManager->doWeCacheGuildMembers()) {
546 GuildMembers::insertGuildMember(dataPackage.guildMember);
547 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.guildMember.guildId }).get();
548 guild.members.push_back(dataPackage.guildMember.id);
549 guild.memberCount++;
550 Guilds::insertGuild(guild);
551 }
552 co_return;
553 }
554
555 CoRoutine<void> EventHandler::onGuildMemberRemove(OnGuildMemberRemoveData dataPackage) {
556 co_await NewThreadAwaitable<void>();
557 if (EventHandler::configManager->doWeCacheGuildMembers()) {
558 GuildMember guildMember = GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = dataPackage.user.id, .guildId = dataPackage.guildId }).get();
559 std::string globalId = std::to_string(guildMember.guildId) + " + " + std::to_string(guildMember.id);
560 GuildMembers::removeGuildMember(guildMember);
561 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.guildId }).get();
562 for (uint64_t x = 0; x < guild.members.size(); x++) {
563 if (guild.members[x] == dataPackage.user.id) {
564 guild.members.erase(guild.members.begin() + x);
565 }
566 }
567 guild.memberCount--;
568 Guilds::insertGuild(guild);
569 }
570 co_return;
