131 void DiscordCoreClient::runBot() {
132 if (!this->instantiateWebSockets()) {
133 Globals::doWeQuit.store(true);
134 return;
135 }
136 while (!Globals::doWeQuit.load()) {
137 std::this_thread::sleep_for(1ms);
138 }
139 if (this->baseSocketAgentMap.contains(std::to_string(this->configManager.getStartingShard())) &&
140 this->baseSocketAgentMap[std::to_string(this->configManager.getStartingShard())]->getTheTask()) {
141 if (this->baseSocketAgentMap[std::to_string(this->configManager.getStartingShard())]->getTheTask()->joinable()) {
142 this->baseSocketAgentMap[std::to_string(this->configManager.getStartingShard())]->getTheTask()->join();
143 }
144 }
145 }

161 bool DiscordCoreClient::instantiateWebSockets() {
162 GatewayBotData gatewayData = this->getGateWayBot();
163 if (gatewayData.url == "") {
164 if (this->configManager.doWePrintGeneralErrorMessages()) {
165 cout << shiftToBrightRed()
166 << "Failed to collect the connection URL! Closing! Did you remember to "
167 "properly set your bot token?"
168 << reset() << endl
169 << endl;
170 }
171 std::this_thread::sleep_for(5s);
172 return false;
173 }
174 if (this->configManager.getStartingShard() + this->configManager.getShardCountForThisProcess() > this->configManager.getTotalShardCount()) {
175 if (this->configManager.doWePrintGeneralErrorMessages()) {
176 cout << shiftToBrightRed() << "Your sharding options are incorrect! Please fix it!" << reset() << endl << endl;
177 }
178 std::this_thread::sleep_for(5s);
179 return false;
180 }
181 auto baseSocketAgentCount = static_cast<int32_t>(std::thread::hardware_concurrency()) > this->configManager.getTotalShardCount()
182 ? this->configManager.getTotalShardCount()
183 : static_cast<int32_t>(std::thread::hardware_concurrency());
184 int32_t shardsPerBaseSocketAgent{ static_cast<int32_t>(floor(static_cast<float>(this->configManager.getTotalShardCount()) / static_cast<float>(baseSocketAgentCount))) };
185 int32_t leftOverShards{ this->configManager.getTotalShardCount() - (shardsPerBaseSocketAgent * baseSocketAgentCount) };
186
187 std::vector<int32_t> shardsPerBaseSocketAgentVect{};
188 for (int32_t x = 0; x < baseSocketAgentCount; x++) {
189 int32_t newShardAmount{};
190 shardsPerBaseSocketAgentVect.push_back(shardsPerBaseSocketAgent);
191 if (leftOverShards == 0) {
192 continue;
193 }
194 newShardAmount = static_cast<int32_t>(ceil(static_cast<float>(leftOverShards) / static_cast<float>(std::thread::hardware_concurrency())));
195 shardsPerBaseSocketAgentVect[x] += newShardAmount;
196
197 if (x == static_cast<int32_t>(std::thread::hardware_concurrency()) - 1) {
198 shardsPerBaseSocketAgentVect[0] += leftOverShards;
199 }
200 leftOverShards -= newShardAmount;
201 }
202 int32_t currentShard{ 0 };
203 if (this->configManager.getConnectionAddress() == "") {
204 this->configManager.setConnectionAddress(gatewayData.url.substr(gatewayData.url.find("wss://") + std::string("wss://").size()));
205 }
206 if (this->configManager.getConnectionPort() == "") {
207 this->configManager.setConnectionPort("443");
208 }
209
210 for (int32_t x = 0; x < shardsPerBaseSocketAgentVect.size(); x++) {
211 auto thePtr = std::make_unique<DiscordCoreInternal::BaseSocketAgent>(this, &Globals::doWeQuit, x);
212 this->baseSocketAgentMap[std::to_string(x)] = std::move(thePtr);
213 for (int32_t y = 0; y < shardsPerBaseSocketAgentVect[x]; y++) {
214 if (this->configManager.doWePrintGeneralSuccessMessages()) {
215 cout << shiftToBrightBlue() << "Connecting Shard " + std::to_string(currentShard + 1) << " of " << this->configManager.getShardCountForThisProcess()
216 << std::string(" Shards for this process. (") + std::to_string(currentShard + 1) + " of " + std::to_string(this->configManager.getTotalShardCount()) +
217 std::string(" Shards total across all processes)")
218 << reset() << endl
219 << endl;
220 }
221 ConnectionPackage theData{};
222 theData.currentShard = currentShard;
223 theData.currentBaseSocketAgent = x;
224 this->baseSocketAgentMap[std::to_string(x)]->connect(theData);
225 currentShard++;
226 }
227 }
228 this->currentUser = BotUser{ Users::getCurrentUserAsync().get(), this->baseSocketAgentMap[std::to_string(this->configManager.getStartingShard())].get() };
229 for (auto& value: this->configManager.getFunctionsToExecute()) {
230 if (value.repeated) {
231 TimeElapsedHandlerNoArgs onSend = [=, this]() -> void {
232 value.function(this);
233 };
234 ThreadPool::storeThread(onSend, value.intervalInMs);
235 } else {
236 TimeElapsedHandler<void*> onSend = [=, this](void*) -> void {
237 value.function(this);
238 };
239 ThreadPool::executeFunctionAfterTimePeriod(onSend, value.intervalInMs, false, static_cast<void*>(&onSend));
240 }
241 }
242 if (this->configManager.doWePrintGeneralSuccessMessages()) {
243 cout << shiftToBrightGreen() << "All of the shards are connected for the current process!" << reset() << endl << endl;
244 }
245 return true;
246 }
