474 void EventHandler::onChannelCreation(OnChannelCreationData dataPackage) {
475 if (EventHandler::configManager->doWeCacheChannels()) {
476 Channels::insertChannel(dataPackage.channel);
477 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.channel.guildId }).get();
478 guild.channels.push_back(dataPackage.channel.id);
479 Guilds::insertGuild(guild);
480 }
481 }
482
483 void EventHandler::onChannelUpdate(OnChannelUpdateData dataPackage) {
484 if (EventHandler::configManager->doWeCacheChannels()) {
485 Channels::insertChannel(dataPackage.channelNew);
486 }
487 }
488
489 void EventHandler::onChannelDeletion(OnChannelDeletionData dataPackage) {
490 if (EventHandler::configManager->doWeCacheChannels()) {
491 Channels::removeChannel(dataPackage.channel.id);
492 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.channel.guildId }).get();
493 for (uint64_t x = 0; x < guild.channels.size(); x++) {
494 if (guild.channels[x] == dataPackage.channel.id) {
495 guild.channels.erase(guild.channels.begin() + x);
496 }
497 }
498 Guilds::insertGuild(guild);
499 }
500 }

514 void EventHandler::onGuildDeletion(OnGuildDeletionData dataPackage) {
515 if (EventHandler::configManager->doWeCacheGuilds()) {
516 for (auto& value: dataPackage.guild.members) {
517 GuildMember guildMember = GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = value, .guildId = dataPackage.guild.id }).get();
518 GuildMembers::removeGuildMember(guildMember);
519 }
520 for (auto& value: dataPackage.guild.channels) {
521 Channels::removeChannel(value);
522 }
523 for (auto& value: dataPackage.guild.roles) {
524 Roles::removeRole(value);
525 }
526 Guilds::removeGuild(dataPackage.guild.id);
527 }
528 }
529
530 void EventHandler::onGuildMemberAdd(OnGuildMemberAddData dataPackage) {
531 if (EventHandler::configManager->doWeCacheGuildMembers()) {
532 GuildMembers::insertGuildMember(dataPackage.guildMember);
533 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.guildMember.guildId }).get();
534 guild.members.push_back(dataPackage.guildMember.id);
535 guild.memberCount++;
536 Guilds::insertGuild(guild);
537 }
538 }
539
540 void EventHandler::onGuildMemberRemove(OnGuildMemberRemoveData dataPackage) {
541 if (EventHandler::configManager->doWeCacheGuildMembers()) {
542 GuildMember guildMember = GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = dataPackage.user.id, .guildId = dataPackage.guildId }).get();
543 std::string globalId = std::to_string(guildMember.guildId) + " + " + std::to_string(guildMember.id);
544 GuildMembers::removeGuildMember(guildMember);
545 GuildData guild = Guilds::getCachedGuildAsync({ dataPackage.guildId }).get();
546 for (uint64_t x = 0; x < guild.members.size(); x++) {
547 if (guild.members[x] == dataPackage.user.id) {
548 guild.members.erase(guild.members.begin() + x);
549 }
550 }
551 guild.memberCount--;
552 Guilds::insertGuild(guild);
553 }
554 }
