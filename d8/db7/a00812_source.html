57 if (this->content != "") {

232 CoRoutine<WebHook> WebHooks::createWebHookAsync(CreateWebHookData dataPackage) {
233 DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Post_Webhook };
234 co_await NewThreadAwaitable<WebHook>();
235 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
236 workload.relativePath = "/channels/" + std::to_string(dataPackage.channelId) + "/webhooks";
237 workload.callStack = "WebHooks::createWebHookAsync()";
238 nlohmann::json responseData = { { "name", dataPackage.name } };
239 if (dataPackage.avatar.size() > 0) {
240 responseData.update({ { "avatar", dataPackage.avatar } });
241 }
242 workload.content = responseData.dump();
243 co_return WebHooks::httpsClient->submitWorkloadAndGetResult<WebHook>(workload);
244 }

340 CoRoutine<Message> WebHooks::executeWebHookAsync(ExecuteWebHookData dataPackage) {
341 DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Post_Execute_Webhook };
342 co_await NewThreadAwaitable<Message>();
343 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
344 workload.relativePath = "/webhooks/" + std::to_string(dataPackage.webHookId) + "/" + dataPackage.webhookToken;
345 workload.callStack = "WebHooks::executeWebHookAsync()";
346 if (dataPackage.wait) {
347 workload.relativePath += "?wait=true";
348 if (dataPackage.threadId != 0) {
349 workload.relativePath += "&thread_id=" + std::to_string(dataPackage.threadId);
350 }
351 }
352 if (dataPackage.threadId != 0) {
353 workload.relativePath += "?thread_id=" + std::to_string(dataPackage.threadId);
354 }
355 if (dataPackage.files.size() > 0) {
356 workload.payloadType = DiscordCoreInternal::PayloadType::Multipart_Form;
357 workload.content = constructMultiPartData(std::string{ dataPackage }, dataPackage.files);
358 } else {
359 workload.content = dataPackage;
360 }
361 co_return WebHooks::httpsClient->submitWorkloadAndGetResult<Message>(workload);
362 }
