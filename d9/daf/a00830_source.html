26 CoRoutine<InputEventData> InputEvents::respondToInputEventAsync(RespondToInputEventData dataPackage) {
27 co_await NewThreadAwaitable<InputEventData>();
28 if (dataPackage.type == InputEventResponseType::Unset) {
29 throw std::runtime_error("InputEvents::respondToInputEventAsync() Error: Please set an "
30 "input-event-response-type!\n\n");
31 }
32 if (dataPackage.eventType == InteractionType::Message_Component) {
33 CreateInteractionResponseData dataPackage02{ dataPackage };
34 if (dataPackage.type == InputEventResponseType::Deferred_Response) {
35 dataPackage02.data.type = InteractionCallbackType::Deferred_Update_Message;
36 } else {
37 dataPackage02.data.type = InteractionCallbackType::Update_Message;
38 }
39 InputEventData newEvent = InputEvents::respondToInputEvent(dataPackage02);
40 if (dataPackage.type == InputEventResponseType::Interaction_Response || dataPackage.type == InputEventResponseType::Ephemeral_Interaction_Response ||
41 dataPackage.type == InputEventResponseType::Edit_Interaction_Response) {
42 newEvent.responseType = InputEventResponseType::Edit_Interaction_Response;
43 } else if (dataPackage.type == InputEventResponseType::Follow_Up_Message || dataPackage.type == InputEventResponseType::Ephemeral_Follow_Up_Message ||
44 dataPackage.type == InputEventResponseType::Edit_Follow_Up_Message) {
45 newEvent.responseType = InputEventResponseType::Edit_Follow_Up_Message;
46 }
47 co_return newEvent;
48 } else if (dataPackage.eventType == InteractionType::Application_Command_Autocomplete) {
49 CreateInteractionResponseData dataPackage02{ dataPackage };
50 dataPackage02.data.type = InteractionCallbackType::Application_Command_Autocomplete_Result;
51 InputEventData newEvent = InputEvents::respondToInputEvent(dataPackage02);
52 newEvent.responseType = InputEventResponseType::Application_Command_AutoComplete_Result;
53 co_return newEvent;
54 }
55 switch (dataPackage.type) {
56 case InputEventResponseType::Ephemeral_Deferred_Response: {
57 auto dataPackage02 = CreateDeferredInteractionResponseData{ dataPackage };
58 dataPackage02.data.data.flags = 64;
59 co_return InputEvents::respondToInputEvent(dataPackage02);
60 }
61 case InputEventResponseType::Deferred_Response: {
62 CreateDeferredInteractionResponseData dataPackage02{ dataPackage };
63 co_return InputEvents::respondToInputEvent(dataPackage02);
64 }
65 case InputEventResponseType::Interaction_Response: {
66 CreateInteractionResponseData dataPackage02{ dataPackage };
67 co_return InputEvents::respondToInputEvent(dataPackage02);
68 }
69 case InputEventResponseType::Edit_Interaction_Response: {
70 EditInteractionResponseData dataPackage02{ dataPackage };
71 co_return InputEvents::respondToInputEvent(dataPackage02);
72 }
73 case InputEventResponseType::Ephemeral_Interaction_Response: {
74 CreateEphemeralInteractionResponseData dataPackage02{ dataPackage };
75 co_return InputEvents::respondToInputEvent(dataPackage02);
76 }
77 case InputEventResponseType::Follow_Up_Message: {
78 CreateFollowUpMessageData dataPackage02{ dataPackage };
79 co_return InputEvents::respondToInputEvent(dataPackage02);
80 }
81 case InputEventResponseType::Edit_Follow_Up_Message: {
82 EditFollowUpMessageData dataPackage02{ dataPackage };
83 co_return InputEvents::respondToInputEvent(dataPackage02);
84 }
85 case InputEventResponseType::Ephemeral_Follow_Up_Message: {
86 CreateEphemeralFollowUpMessageData dataPackage02{ dataPackage };
87 co_return InputEvents::respondToInputEvent(dataPackage02);
88 }
89 case InputEventResponseType::Unset: {
90 break;
91 }
92 default: {
93 co_return InputEventData();
94 }
95 }
96 co_return InputEventData();
97 }

99 CoRoutine<void> InputEvents::deleteInputEventResponseAsync(InputEventData& dataPackage, int32_t timeDelayNew) {
100 InputEventData newPackage = dataPackage;
101 co_await NewThreadAwaitable<void>();
102 if (newPackage.responseType == InputEventResponseType::Follow_Up_Message || newPackage.responseType == InputEventResponseType::Edit_Follow_Up_Message) {
103 RespondToInputEventData dataPackageNew{ newPackage };
104 DeleteFollowUpMessageData dataPackageNewer{ dataPackageNew };
105 dataPackageNewer.timeDelay = timeDelayNew;
106 Interactions::deleteFollowUpMessageAsync(dataPackageNewer).get();
107 } else if (newPackage.responseType == InputEventResponseType::Interaction_Response || newPackage.responseType == InputEventResponseType::Edit_Interaction_Response ||
108 newPackage.responseType == InputEventResponseType::Deferred_Response) {
109 RespondToInputEventData dataPackageNew{ newPackage };
110 DeleteInteractionResponseData dataPackageNewer{ dataPackageNew };
111 dataPackageNewer.timeDelay = timeDelayNew;
112 Interactions::deleteInteractionResponseAsync(dataPackageNewer).get();
113 }
114 co_return;
115 }

117 InputEventData InputEvents::respondToInputEvent(CreateDeferredInteractionResponseData& dataPackage) {
118 CreateInteractionResponseData dataPackageNew{ dataPackage };
119 Interactions::createInteractionResponseAsync(dataPackageNew).get();
120 InputEventData dataPackageNewer{};
121 dataPackageNewer.responseType = InputEventResponseType::Deferred_Response;
122 dataPackageNewer.interactionData->applicationId = dataPackage.interactionPackage.applicationId;
123 dataPackageNewer.interactionData->token = dataPackage.interactionPackage.interactionToken;
124 dataPackageNewer.interactionData->id = dataPackage.interactionPackage.interactionId;
125 return dataPackageNewer;
126 }
