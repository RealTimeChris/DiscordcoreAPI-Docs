81 /// For modifying a GuildMember's values. \brief For modifying a GuildMember's values.
82 struct DiscordCoreAPI_Dll ModifyGuildMemberData {
83 TimeStamp communicationDisabledUntil{ "" };///< When the user's timeout will expire and the user will be able to communicate in the guild again.
84 std::vector<Snowflake> roleIds{};///< A collection of Role id's to be applied to them.
85 Snowflake newVoiceChannelId{};///< The new voice Channel to move them into.
86 Snowflake currentChannelId{};///< The current voice Channel, if applicaple.
87 Snowflake guildMemberId{};///< The user id of the desired Guild memeber.
88 Snowflake guildId{};///< The id of the Guild for which you would like to modify a member.
89 std::string reason{};///< Reason for modifying this GuildMember.
90 std::string nick{};///< Their new display/nick name.
91 bool mute{ false };///< Whether or not to mute them in voice.
92 bool deaf{ false };///< Whether or not to deafen them, in voice.
93
94 operator std::string() {
95 nlohmann::json data{};
96 data["nick"] = this->nick;
97 data["communication_disabled_until"] = std::string(this->communicationDisabledUntil);
98 if (this->roleIds.size() == 0) {
99 data["roles"] = nullptr;
100 } else {
101 nlohmann::json roleIdArray{};
102 for (auto& value: this->roleIds) {
103 roleIdArray.push_back(value);
104 }
105 data["roles"] = roleIdArray;
106 }
107 if (this->newVoiceChannelId != 0) {
108 data["channel_id"] = std::to_string(this->newVoiceChannelId);
109 data["mute"] = this->mute;
110 data["deaf"] = this->deaf;
111 }
112 return data.dump();
113 }
114 };
