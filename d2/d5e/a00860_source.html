61 data["message"]["flags"] = this->message.flags;
62 data["name"] = this->name;
63 data["auto_archive_duration"] = this->autoArchiveDuration;
64 data["rate_limit_per_user"] = this->rateLimitPerUser;

81 CoRoutine<Thread> Threads::startThreadWithMessageAsync(StartThreadWithMessageData dataPackage) {
82 DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Post_Thread_With_Message };
83 co_await NewThreadAwaitable<Thread>();
84 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
85 workload.relativePath = "/channels/" + std::to_string(dataPackage.channelId) + "/messages/" + std::to_string(dataPackage.messageId) + "/threads";
86 workload.content = dataPackage;
87 workload.callStack = "Threads::startThreadWithMessageAsync()";
88 if (dataPackage.reason != "") {
89 workload.headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
90 }
91 co_return Threads::httpsClient->submitWorkloadAndGetResult<Thread>(workload);
92 }
