40 CoRoutine<void> CommandController::checkForAndRunCommand(CommandData commandData) {
41 co_await NewThreadAwaitable<void>();
42 try {
43 std::unique_ptr<BaseFunction> functionPointer{ this->getCommand(convertToLowerCase(commandData.commandName)) };
44 if (functionPointer == nullptr) {
45 co_return;
46 }
47 BaseFunctionArguments theArgs{ commandData, this->discordCoreClient };
48 functionPointer->execute(theArgs);
49 co_return;
50 } catch (...) {
51 reportException("CommandController::checkForAndRunCommand()");
52 }
53 co_return;
54 }

56 std::unique_ptr<BaseFunction> CommandController::getCommand(const std::string& commandName) {
57 std::string functionName{};
58 bool isItFound{ false };
59 if (commandName.size() > 0) {
60 for (auto const& [keyFirst, value]: Globals::functions) {
61 for (auto& key: keyFirst) {
62 if (key.find(convertToLowerCase(commandName)) != std::string::npos) {
63 isItFound = true;
64 functionName = convertToLowerCase(commandName.substr(0, key.length()));
65 break;
66 }
67 }
68 }
69 }
70 if (isItFound) {
71 std::unique_ptr<BaseFunction> newValue = this->createFunction(functionName);
72 return newValue;
73 }
74 return nullptr;
75 }

77 std::unique_ptr<BaseFunction> CommandController::createFunction(const std::string& functionName) {
78 for (auto& [key01, value01]: Globals::functions) {
79 for (auto& value02: key01) {
80 if (functionName == value02) {
81 return Globals::functions[key01]->create();
82 }
83 }
84 }
85 return nullptr;
86 }
