185  CoRoutine<EmojiData> Reactions::createGuildEmojiAsync(CreateGuildEmojiData dataPackage) {
186  DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Post_Guild_Emoji };
187  co_await NewThreadAwaitable<EmojiData>();
188  workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
189  std::ifstream fin(dataPackage.imageFilePath, std::ios::binary);
190  fin.seekg(0, std::ios::end);
191  std::string data{};
192  data.resize(fin.tellg());
193  fin.seekg(0, std::ios::beg);
194  fin.read(data.data(), data.size());
195  std::string newerFile = base64Encode(data);
196  switch (dataPackage.type) {
197  case ImageType::Jpg: {
198  dataPackage.imageDataFinal = "data:image/jpeg;base64,";
199  dataPackage.imageDataFinal.insert(dataPackage.imageDataFinal.end(), newerFile.begin(), newerFile.end());
200  break;
201  }
202  case ImageType::Png: {
203  dataPackage.imageDataFinal = "data:image/png;base64,";
204  dataPackage.imageDataFinal.insert(dataPackage.imageDataFinal.end(), newerFile.begin(), newerFile.end());
205  break;
206  }
207  case ImageType::Gif: {
208  dataPackage.imageDataFinal = "data:image/gif;base64,";
209  dataPackage.imageDataFinal.insert(dataPackage.imageDataFinal.end(), newerFile.begin(), newerFile.end());
210  break;
211  }
212  }
213  workload.relativePath = "/guilds/" + std::to_string(dataPackage.guildId) + "/emojis";
214  workload.content = dataPackage;
215  workload.callStack = "Reactions::createGuildEmojiAsync()";
216  if (dataPackage.reason != "") {
217  workload.headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
218  }
219  co_return Reactions::httpsClient->submitWorkloadAndGetResult<EmojiData>(workload);
220  }
