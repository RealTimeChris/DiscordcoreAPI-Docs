178 CoRoutine<EmojiData> Reactions::createGuildEmojiAsync(CreateGuildEmojiData dataPackage) {
179 DiscordCoreInternal::HttpsWorkloadData workload{};
180 workload.thisWorkerId = DiscordCoreInternal::HttpsWorkloadData::getAndIncrementWorkloadId(DiscordCoreInternal::HttpsWorkloadType::Post_Guild_Emoji);
181 co_await NewThreadAwaitable<EmojiData>();
182 workload.workloadType = DiscordCoreInternal::HttpsWorkloadType::Post_Guild_Emoji;
183 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
184 std::ifstream fin(dataPackage.imageFilePath, std::ios::binary);
185 fin.seekg(0, std::ios::end);
186 std::string data{};
187 data.resize(fin.tellg());
188 fin.seekg(0, std::ios::beg);
189 fin.read(data.data(), data.size());
190 std::string newerFile = base64Encode(data);
191 switch (dataPackage.type) {
192 case ImageType::Jpg: {
193 dataPackage.imageDataFinal = "data:image/jpeg;base64,";
194 dataPackage.imageDataFinal.insert(dataPackage.imageDataFinal.end(), newerFile.begin(), newerFile.end());
195 break;
196 }
197 case ImageType::Png: {
198 dataPackage.imageDataFinal = "data:image/png;base64,";
199 dataPackage.imageDataFinal.insert(dataPackage.imageDataFinal.end(), newerFile.begin(), newerFile.end());
200 break;
201 }
202 case ImageType::Gif: {
203 dataPackage.imageDataFinal = "data:image/gif;base64,";
204 dataPackage.imageDataFinal.insert(dataPackage.imageDataFinal.end(), newerFile.begin(), newerFile.end());
205 break;
206 }
207 }
208 workload.relativePath = "/guilds/" + std::to_string(dataPackage.guildId) + "/emojis";
209 workload.content = dataPackage;
210 workload.callStack = "Reactions::createGuildEmojiAsync()";
211 if (dataPackage.reason != "") {
212 workload.headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
213 }
214 co_return Reactions::httpsClient->submitWorkloadAndGetResult<EmojiData>(workload);
215 }
