126 CoRoutine<GuildMember> GuildMembers::getGuildMemberAsync(GetGuildMemberData dataPackage) {
127 DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Get_Guild_Member };
128 co_await NewThreadAwaitable<GuildMember>();
129 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Get;
130 workload.relativePath = "/guilds/" + std::to_string(dataPackage.guildId) + "/members/" + std::to_string(dataPackage.guildMemberId);
131 workload.callStack = "GuildMembers::getGuildMemberAsync()";
132 GuildMember guildMember = GuildMembers::httpsClient->submitWorkloadAndGetResult<GuildMember>(workload);
133 guildMember = GuildMembers::getCachedGuildMemberAsync(dataPackage).get();
134 guildMember.guildId = dataPackage.guildId;
135 co_return guildMember;
136 }

138 CoRoutine<GuildMemberData> GuildMembers::getCachedGuildMemberAsync(GetGuildMemberData dataPackage) {
139 co_await NewThreadAwaitable<GuildMemberData>();
140 GuildMemberId theKey{};
141 theKey.guildId = dataPackage.guildId;
142 theKey.guildMemberId = dataPackage.guildMemberId;
143 std::shared_lock theLock{ GuildMembers::theMutex };
144 if (!GuildMembers::cache->contains(theKey)) {
145 theLock.unlock();
146 auto theGuildMember = GuildMembers::getGuildMemberAsync(dataPackage).get();
147 GuildMembers::insertGuildMember(theGuildMember);
148 co_return theGuildMember;
149 } else {
150 co_return *(*GuildMembers::cache)[theKey];
151 }
152 }

238 CoRoutine<GuildMember> GuildMembers::timeoutGuildMemberAsync(TimeoutGuildMemberData dataPackage) {
239 co_await NewThreadAwaitable<GuildMember>();
240 GuildMember guildMember = GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = dataPackage.guildMemberId, .guildId = dataPackage.guildId }).get();
241 ModifyGuildMemberData dataPackage01{};
242 dataPackage01.deaf = getBool<int8_t, GuildMemberFlags>(guildMember.flags, GuildMemberFlags::Deaf);
243 dataPackage01.guildId = guildMember.guildId;
244 dataPackage01.guildMemberId = guildMember.id;
245 dataPackage01.mute = getBool<int8_t, GuildMemberFlags>(guildMember.flags, GuildMemberFlags::Mute);
246 dataPackage01.roleIds = guildMember.roles;
247 dataPackage01.nick = guildMember.nick;
248 dataPackage01.reason = dataPackage.reason;
249 switch (dataPackage.numOfMinutesToTimeoutFor) {
250 case TimeoutDurations::Day: {
251 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(0, 0, 1, 0, 0);
252 break;
253 }
254 case TimeoutDurations::Five_Minutes: {
255 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(5, 0, 0, 0, 0);
256 break;
257 }
258 case TimeoutDurations::Hour: {
259 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(0, 1, 0, 0, 0);
260 break;
261 }
262 case TimeoutDurations::Ten_Minutes: {
263 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(10, 0, 0, 0, 0);
264 break;
265 }
266 case TimeoutDurations::Week: {
267 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(0, 0, 7, 0, 0);
268 break;
269 }
270 case TimeoutDurations::Minute: {
271 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(1, 0, 0, 0, 0);
272 break;
273 }
274 case TimeoutDurations::None: {
275 dataPackage01.communicationDisabledUntil = TimeStamp::getFutureISO8601TimeStamp(0, 0, 0, 0, 0);
276 break;
277 }
278 }
279 GuildMember newGuildMember = GuildMembers::modifyGuildMemberAsync(dataPackage01).get();
280 co_return newGuildMember;
281 }

283 void GuildMembers::insertGuildMember(GuildMemberData guildMember) {
284 std::unique_lock theLock{ GuildMembers::theMutex };
285 if (guildMember.id == 0) {
286 return;
287 }
288 auto newCache = std::make_unique<std::map<GuildMemberId, std::unique_ptr<GuildMemberData>>>();
289 for (auto& [key, value]: *GuildMembers::cache) {
290 (*newCache)[key] = std::move(value);
291 }
292 GuildMemberId theKey{};
293 if (GuildMembers::configManager->doWeCacheGuildMembers()) {
294 theKey.guildId = guildMember.guildId;
295 theKey.guildMemberId = guildMember.id;
296 (*newCache)[theKey] = std::make_unique<GuildMemberData>(guildMember);
297 }
298 GuildMembers::cache.reset(nullptr);
299 GuildMembers::cache = std::move(newCache);
300 }

302 void GuildMembers::removeGuildMember(GuildMemberData& guildMember) {
303 std::unique_lock theLock{ GuildMembers::theMutex };
304 GuildMemberId theKey{};
305 theKey.guildId = guildMember.guildId;
306 theKey.guildMemberId = guildMember.id;
307 GuildMembers::cache->erase(theKey);
308 };
