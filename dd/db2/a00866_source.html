159 CoRoutine<UserData> Users::getCachedUserAsync(GetUserData dataPackage) {
160 co_await NewThreadAwaitable<UserData>();
161 std::shared_lock<std::shared_mutex> theLock{ Users::theMutex };
162 if (!Users::cache->contains(dataPackage.userId)) {
163 theLock.unlock();
164 auto theUser = getUserAsync(dataPackage).get();
165 Users::insertUser(theUser);
166 co_return theUser;
167
168 } else {
169 co_return *(*Users::cache)[dataPackage.userId];
170 }
171 }

235 void Users::insertUser(UserData user) {
236 std::unique_lock<std::shared_mutex> theLock{ Users::theMutex };
237 if (user.id == 0) {
238 return;
239 }
240 auto newCache = std::make_unique<std::unordered_map<uint64_t, std::unique_ptr<UserData>>>();
241 for (auto& [key, value]: *Users::cache) {
242 (*newCache)[key] = std::move(value);
243 }
244 if (Users::configManager->doWeCacheUsers()) {
245 (*newCache)[user.id] = std::make_unique<UserData>(user);
246 }
247 Users::cache.reset(nullptr);
248 Users::cache = std::move(newCache);
249 }
