26 UpdatePresenceData::operator nlohmann::json() {

103 void BotUser::updateVoiceStatus(UpdateVoiceStateData& dataPackage) {
104 nlohmann::json payload = dataPackage;
105 std::string theString{};
106 if (this->baseSocketAgent) {
107 int32_t theIndex{};
108 for (auto& [key, value]: this->baseSocketAgent->sslShards) {
109 theIndex = key;
110 break;
111 }
112 std::string theString{};
113 if (static_cast<DiscordCoreInternal::WebSocketSSLShard*>(this->baseSocketAgent->sslShards[theIndex].get())->dataOpCode ==
114 DiscordCoreInternal::WebSocketOpCode::Op_Binary) {
115 this->baseSocketAgent->sslShards[theIndex]->stringifyJsonData(payload, theString, DiscordCoreInternal::WebSocketOpCode::Op_Binary);
116 } else {
117 this->baseSocketAgent->sslShards[theIndex]->stringifyJsonData(payload, theString, DiscordCoreInternal::WebSocketOpCode::Op_Text);
118 }
119 this->baseSocketAgent->sendMessage(theString, static_cast<DiscordCoreInternal::WebSocketSSLShard*>(this->baseSocketAgent->sslShards[theIndex].get()), true);
120 }

197 CoRoutine<UserData> Users::getCachedUserAsync(GetUserData dataPackage) {
198 co_await NewThreadAwaitable<UserData>();
199 std::shared_lock theLock{ Users::theMutex };
200 if (!Users::cache->contains(dataPackage.userId)) {
201 theLock.unlock();
202 auto theUser = getUserAsync(dataPackage).get();
203 Users::insertUser(theUser);
204 co_return theUser;
205
206 } else {
207 co_return *(*Users::cache)[dataPackage.userId];
208 }
209 }

265 void Users::insertUser(UserData user) {
266 std::unique_lock theLock{ Users::theMutex };
267 if (user.id == 0) {
268 return;
269 }
270 auto newCache = std::make_unique<std::unordered_map<Snowflake, std::unique_ptr<UserData>>>();
271 for (auto& [key, value]: *Users::cache) {
272 (*newCache)[key] = std::move(value);
273 }
274 if (Users::configManager->doWeCacheUsers()) {
275 (*newCache)[user.id] = std::make_unique<UserData>(user);
276 }
277 Users::cache.reset(nullptr);
278 Users::cache = std::move(newCache);
279 }
