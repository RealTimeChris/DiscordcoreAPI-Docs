49  void ErlPacker::singleValueJsonToETF(ErlPackBuffer& buffer, const nlohmann::json& jsonData) {
50  if (jsonData.is_array()) {
51  uint32_t length = static_cast<uint32_t>(jsonData.size());
52  if (length == 0) {
53  ErlPacker::appendNilExt(buffer);
54  } else {
55  if (length > std::numeric_limits<uint32_t>::max() - 1) {
56  throw ErlPackError{ "ErlPacker::singleValueJsonToETF() Error: List too large for ETF.\n\n" };
57  }
58  }
59  ErlPacker::appendListHeader(buffer, length);
60  for (uint64_t index = 0; index < length; ++index) {
61  ErlPacker::singleValueJsonToETF(buffer, jsonData[index]);
62  }
63  ErlPacker::appendNilExt(buffer);
64  } else if (jsonData.is_object()) {
65  uint32_t length = static_cast<uint32_t>(jsonData.size());
66  if (length > std::numeric_limits<uint32_t>::max() - 1) {
67  throw ErlPackError{ "ErlPacker::singleValueJsonToETF() Error: Map too large for ETF.\n\n" };
68  }
69  ErlPacker::appendMapHeader(buffer, length);
70  for (auto n = jsonData.begin(); n != jsonData.end(); ++n) {
71  nlohmann::json jstr = n.key();
72  ErlPacker::singleValueJsonToETF(buffer, jstr);
73  ErlPacker::singleValueJsonToETF(buffer, n.value());
74  }
75  } else if (jsonData.is_number_integer()) {
76  uint64_t numberOld = jsonData.get<uint64_t>();
77  if (numberOld >= 0 && numberOld <= 127) {
78  uint8_t number = jsonData.get<uint8_t>();
79  ErlPacker::appendSmallIntegerExt(buffer, number);
80  } else if (jsonData.is_number_unsigned() && (numberOld >= std::numeric_limits<uint32_t>::max() - static_cast<size_t>(1))) {
81  uint64_t number = jsonData.get<uint64_t>();
82  ErlPacker::appendUnsignedLongLong(buffer, number);
83  } else {
84  uint32_t number = jsonData.get<uint32_t>();
85  ErlPacker::appendIntegerExt(buffer, number);
86  }
87  } else if (jsonData.is_boolean()) {
88  if (jsonData.get<bool>()) {
89  ErlPacker::appendTrue(buffer);
90  } else {
91  ErlPacker::appendFalse(buffer);
92  }
93  } else if (jsonData.is_string()) {
94  std::string newString = jsonData.get<std::string>();
95  std::string newVector{};
96  newVector.insert(newVector.begin(), newString.begin(), newString.end());
97  uint32_t newValue = static_cast<uint32_t>(newVector.size());
98  ErlPacker::appendBinaryExt(buffer, newVector, newValue);
99  } else if (jsonData.is_number_float()) {
100  double newValue = jsonData.get<double>();
101  ErlPacker::appendFloatExt(buffer, newValue);
102  } else if (jsonData.is_null()) {
103  ErlPacker::appendNil(buffer);
104  }
105  }

158  void ErlPacker::appendUnsignedLongLong(ErlPackBuffer& buffer, uint64_t& value) {
159  std::string bufferNew{};
160  bufferNew.resize(static_cast<uint64_t>(1) + 2 + sizeof(uint64_t));
161  bufferNew[0] = static_cast<uint8_t>(ETFTokenType::Small_Big_Ext);
162  uint8_t bytesToEncode = 0;
163  while (value > 0) {
164  bufferNew[static_cast<size_t>(3) + bytesToEncode] = value & 0xF;
165  value >>= 8;
166  bytesToEncode++;
167  }
168  bufferNew[1] = bytesToEncode;
169  bufferNew[2] = 0;
170  ErlPacker::writeToBuffer(buffer, bufferNew);
171  }

192  template<typename ReturnType> void ErlPacker::readBits(const ErlPackBuffer& buffer, ReturnType& theValue) {
193  const uint8_t byteSize{ 8 };
194  if (buffer.offSet + sizeof(ReturnType) > buffer.buffer.size()) {
195  throw ErlPackError{ "ErlPacker::readBits() Error: readBits() past end of buffer.\n\n" };
196  }
197  ReturnType newValue{ 0 };
198  for (uint64_t x = 0; x < sizeof(ReturnType); x++) {
199  newValue |= static_cast<ReturnType>(static_cast<uint64_t>(buffer.buffer.data()[buffer.offSet + x]) << (x * static_cast<uint64_t>(byteSize)));
200  }
201  buffer.offSet += sizeof(ReturnType);
202  theValue = DiscordCoreAPI::reverseByteOrder(newValue);
203  }

280  nlohmann::json ErlPacker::parseBigint(const ErlPackBuffer& buffer, uint32_t& digits) {
281  uint8_t sign{};
282  readBits(buffer, sign);
283  if (digits > 8) {
284  throw ErlPackError{ "ErlPacker::parseBigint() Error: Integers larger than 8 bytes are not supported.\n\n" };
285  }
286  uint64_t value = 0;
287  uint64_t b = 1;
288  for (uint32_t x = 0; x < digits; ++x) {
289  uint8_t digit{};
290  readBits(buffer, digit);
291  uint64_t digitNew = digit;
292  value += digitNew * b;
293  b <<= 8;
294  }
295  if (digits <= 4) {
296  if (sign == 0) {
297  nlohmann::json jsonData = std::to_string(value);
298  return jsonData;
299  }
300  const bool isSignBitAvailable = (value & 1ull << 31ull) == 0;
301  if (isSignBitAvailable) {
302  nlohmann::json jsonData = std::to_string(-static_cast<int32_t>(value));
303  return jsonData;
304  }
305  }
306  char outBuffer[32] = { 0 };
307  const char* formatString = sign == 0 ? "%llu" : "-%ll";
308  const int32_t res = sprintf(outBuffer, formatString, value);
309  if (res < 0) {
310  throw ErlPackError{ "ErlPacker::parseBigint() Error: Parse big integer failed.\n\n" };
311  }
312  const uint8_t length = static_cast<uint8_t>(res);
313  nlohmann::json jsonData = std::string(outBuffer, length);
314  return jsonData;
315  }

324  nlohmann::json ErlPacker::parseFloatExt(const ErlPackBuffer& buffer) {
325  uint32_t floatLength = 31;
326  std::string floatStr{};
327  nlohmann::json jsonData{};
328  readString(buffer, floatLength, floatStr);
329  if (floatStr.size() == 0) {
330  return jsonData;
331  }
332  double number{};
333  std::vector<char> nullTerminated{};
334  nullTerminated.insert(nullTerminated.begin(), floatStr.begin(), floatStr.begin() + floatLength);
335  auto count = sscanf(nullTerminated.data(), "%lf", &number);
336  if (!count) {
337  return jsonData;
338  }
339  jsonData = number;
340  return jsonData;
341  }

343  nlohmann::json ErlPacker::processAtom(const std::string& atom, uint32_t& length) {
344  nlohmann::json jsonData{};
345  if (atom.size() == 0) {
346  return jsonData;
347  }
348  if (length >= 3 && length <= 5) {
349  if (length == 3 && strncmp(atom.data(), "nil", 3) == 0) {
350  return jsonData;
351  } else if (length == 4 && strncmp(atom.data(), "null", 4) == 0) {
352  return jsonData;
353  } else if (length == 4 && strncmp(atom.data(), "true", 4) == 0) {
354  jsonData = true;
355  return jsonData;
356  } else if (length == 5 && strncmp(atom.data(), "false", 5) == 0) {
357  jsonData = false;
358  return jsonData;
359  }
360  }
361  jsonData = std::string(atom.data(), length);
362  return jsonData;
363  }
