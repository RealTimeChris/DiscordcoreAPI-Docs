156 std::vector<Song> SongAPI::searchForSong(const std::string& searchQuery, const Snowflake& guildId) {
157 auto vector01 = getSoundCloudAPIMap()[guildId]->searchForSong(searchQuery);
158 auto vector02 = getYouTubeAPIMap()[guildId]->searchForSong(searchQuery);
159 int32_t totalLength = static_cast<int32_t>(vector01.size() + vector02.size());
160 std::vector<Song> newVector{};
161 int32_t vector01Used{ 0 };
162 int32_t vector02Used{ 0 };
163 for (int32_t x = 0; x < totalLength; x++) {
164 if ((vector01Used < vector01.size() - 1) && (x % 2 == 0) && vector01.size() > 0) {
165 vector01[vector01Used].type = SongType::SoundCloud;
166 newVector.push_back(vector01[vector01Used]);
167 vector01Used++;
168 } else if (vector02Used < vector02.size() - 1 && vector02.size() > 0) {
169 vector02[vector02Used].type = SongType::YouTube;
170 newVector.push_back(vector02[vector02Used]);
171 vector02Used++;
172 }
173 }
174 return newVector;
175 }
