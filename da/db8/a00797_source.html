56  std::unique_ptr<BaseFunction> CommandController::getCommand(const std::string& commandName) {
57  std::string functionName{};
58  bool isItFound{ false };
59  if (commandName.size() > 0) {
60  for (auto const& [keyFirst, value]: this->functions) {
61  for (auto& key: keyFirst) {
62  if (key.find(convertToLowerCase(commandName)) != std::string::npos) {
63  isItFound = true;
64  functionName = convertToLowerCase(commandName.substr(0, key.length()));
65  break;
66  }
67  }
68  }
69  }
70  if (isItFound) {
71  std::unique_ptr<BaseFunction> newValue = this->createFunction(functionName);
