53 CoRoutine<Sticker> Stickers::getStickerAsync(GetStickerData dataPackage) {
54 DiscordCoreInternal::HttpsWorkloadData workload{};
55 workload.thisWorkerId = DiscordCoreInternal::HttpsWorkloadData::getAndIncrementWorkloadId(DiscordCoreInternal::HttpsWorkloadType::Get_Sticker);
56 co_await NewThreadAwaitable<Sticker>();
57 workload.workloadType = DiscordCoreInternal::HttpsWorkloadType::Get_Sticker;
58 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Get;
59 workload.relativePath = "/stickers/" + dataPackage.stickerId;
60 workload.callStack = "Stickers::getStickerAsync()";
61 co_return Stickers::httpsClient->submitWorkloadAndGetResult<Sticker>(workload);
62 }

86 CoRoutine<Sticker> Stickers::createGuildStickerAsync(CreateGuildStickerData dataPackage) {
87 DiscordCoreInternal::HttpsWorkloadData workload{};
88 workload.thisWorkerId = DiscordCoreInternal::HttpsWorkloadData::getAndIncrementWorkloadId(DiscordCoreInternal::HttpsWorkloadType::Post_Guild_Sticker);
89 co_await NewThreadAwaitable<Sticker>();
90 workload.workloadType = DiscordCoreInternal::HttpsWorkloadType::Post_Guild_Sticker;
91 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
92 workload.relativePath = "/guilds/" + std::to_string(dataPackage.guildId) + "/stickers";
93 nlohmann::json responseData = { { "description", dataPackage.description }, { "name", dataPackage.name }, { "tags", dataPackage.tags }, { "file", dataPackage.file } };
94 workload.content = responseData.dump();
95 workload.callStack = "Stickers::createGuildStickerAsync()";
96 if (dataPackage.reason != "") {
97 workload.headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
98 }
99 co_return Stickers::httpsClient->submitWorkloadAndGetResult<Sticker>(workload);
100 }
