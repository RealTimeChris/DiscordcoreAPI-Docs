141 bool DiscordCoreClient::instantiateWebSockets() {
142 GatewayBotData gatewayData = this->getGateWayBot();
143 if (gatewayData.url == "") {
144 if (this->configManager.doWePrintGeneralErrorMessages()) {
145 std::cout << shiftToBrightRed()
146 << "Failed to collect the connection URL! Closing! Did you remember to "
147 "properly set your bot token?"
148 << reset() << std::endl
149 << std::endl;
150 }
151 std::this_thread::sleep_for(5s);
152 return false;
153 }
154 if (this->configManager.getStartingShard() + this->configManager.getShardCountForThisProcess() > this->configManager.getTotalShardCount()) {
155 if (this->configManager.doWePrintGeneralErrorMessages()) {
156 std::cout << shiftToBrightRed() << "Your sharding options are incorrect! Please fix it!" << reset() << std::endl << std::endl;
157 }
158 std::this_thread::sleep_for(5s);
159 return false;
160 }
161 auto baseSocketAgentCount = static_cast<int32_t>(std::thread::hardware_concurrency()) > this->configManager.getTotalShardCount()
162 ? this->configManager.getTotalShardCount()
163 : static_cast<int32_t>(std::thread::hardware_concurrency());
164 int32_t shardsPerBaseSocketAgent{ static_cast<int32_t>(floor(static_cast<float>(this->configManager.getTotalShardCount()) / static_cast<float>(baseSocketAgentCount))) };
165 int32_t leftOverShards{ this->configManager.getTotalShardCount() - (shardsPerBaseSocketAgent * baseSocketAgentCount) };
166
167 std::vector<int32_t> shardsPerBaseSocketAgentVect{};
168 for (int32_t x = 0; x < baseSocketAgentCount; x++) {
169 int32_t newShardAmount{};
170 shardsPerBaseSocketAgentVect.push_back(shardsPerBaseSocketAgent);
171 if (leftOverShards == 0) {
172 continue;
173 }
174 newShardAmount = static_cast<int32_t>(ceil(static_cast<float>(leftOverShards) / static_cast<float>(std::thread::hardware_concurrency())));
175 shardsPerBaseSocketAgentVect[x] += newShardAmount;
176
177 if (x == static_cast<int32_t>(std::thread::hardware_concurrency()) - 1) {
178 shardsPerBaseSocketAgentVect[0] += leftOverShards;
179 }
180 leftOverShards -= newShardAmount;
181 }
182 int32_t currentShard{ 0 };
183 if (this->configManager.getConnectionAddress() == "") {
184 this->configManager.setConnectionAddress(gatewayData.url.substr(gatewayData.url.find("wss://") + std::string("wss://").size()));
185 }
186 if (this->configManager.getConnectionPort() == "") {
187 this->configManager.setConnectionPort("443");
188 }
189
190 for (int32_t x = 0; x < shardsPerBaseSocketAgentVect.size(); x++) {
191 auto thePtr = std::make_unique<DiscordCoreInternal::BaseSocketAgent>(this, &Globals::doWeQuit, x);
192 this->baseSocketAgentMap[std::to_string(x)] = std::move(thePtr);
193 for (int32_t y = 0; y < shardsPerBaseSocketAgentVect[x]; y++) {
194 if (this->configManager.doWePrintGeneralSuccessMessages()) {
195 std::cout << shiftToBrightBlue() << "Connecting Shard " + std::to_string(currentShard + 1) << " of " << this->configManager.getShardCountForThisProcess()
196 << std::string(" Shards for this process. (") + std::to_string(currentShard + 1) + " of " + std::to_string(this->configManager.getTotalShardCount()) +
197 std::string(" Shards total across all processes)")
198 << reset() << std::endl
199 << std::endl;
200 }
201 ConnectionPackage theData{};
202 theData.currentShard = currentShard;
203 theData.currentBaseSocketAgent = x;
204 this->baseSocketAgentMap[std::to_string(x)]->connect(theData);
205 currentShard++;
206 }
207 }
208 this->currentUser = BotUser{ Users::getCurrentUserAsync().get(), this->baseSocketAgentMap[std::to_string(this->configManager.getStartingShard())].get() };
209 for (auto& value: this->configManager.getFunctionsToExecute()) {
210 TimeElapsedHandler<int64_t> onSend = [=, this](int64_t theArg) -> void {
211 value.function(this);
212 };
213 this->threadIds.push_back(ThreadPool::storeThread(onSend, value.intervalInMs, value.repeated, value.dummyArg));
214 }
215 if (this->configManager.doWePrintGeneralSuccessMessages()) {
216 std::cout << shiftToBrightGreen() << "All of the shards are connected for the current process!" << reset() << std::endl << std::endl;
217 }
218 return true;
219 }
