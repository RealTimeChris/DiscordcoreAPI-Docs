216 co_await NewThreadAwaitable<ChannelData>();
217 std::shared_lock theLock{ Channels::theMutex };
218 if (!Channels::cache->contains(dataPackage.channelId)) {
219 theLock.unlock();
220 auto theChannel = Channels::getChannelAsync(dataPackage).get();
221 Channels::insertChannel(theChannel);

404 void Channels::insertChannel(ChannelData channel) {
405 std::unique_lock theLock{ Channels::theMutex };
406 if (channel.id == 0) {
407 return;
408 }
409 auto newCache = std::make_unique<std::unordered_map<Snowflake, std::unique_ptr<ChannelData>>>();
410 for (auto& [key, value]: *Channels::cache) {
411 (*newCache)[key] = std::move(value);
412 }
413 if (Channels::configManager->doWeCacheChannels()) {
414 (*newCache)[channel.id] = std::make_unique<ChannelData>(channel);
415 }
416 Channels::cache.reset(nullptr);
417 Channels::cache = std::move(newCache);
418 }

420 void Channels::removeChannel(const Snowflake& channelId) {
421 std::unique_lock theLock{ Channels::theMutex };
422 Channels::cache->erase(channelId);
423 };
