106 CoRoutine<ChannelData> Channels::getCachedChannelAsync(GetChannelData dataPackage) {
107 co_await NewThreadAwaitable<ChannelData>();
108 std::shared_lock<std::shared_mutex> theLock{ Channels::theMutex };
109 if (!Channels::cache->contains(dataPackage.channelId)) {
110 theLock.unlock();
111 auto theChannel = Channels::getChannelAsync(dataPackage).get();
112 Channels::insertChannel(theChannel);
113 co_return theChannel;
114 } else {
115 co_return *(*Channels::cache)[dataPackage.channelId];
116 }
117 }

295 void Channels::insertChannel(ChannelData channel) {
296 std::unique_lock<std::shared_mutex> theLock{ Channels::theMutex };
297 if (channel.id == 0) {
298 return;
299 }
300 auto newCache = std::make_unique<std::unordered_map<uint64_t, std::unique_ptr<ChannelData>>>();
301 for (auto& [key, value]: *Channels::cache) {
302 (*newCache)[key] = std::move(value);
303 }
304 if (Channels::configManager->doWeCacheChannels()) {
305 (*newCache)[channel.id] = std::make_unique<ChannelData>(channel);
306 }
307 Channels::cache.reset(nullptr);
308 Channels::cache = std::move(newCache);
309 }

311 void Channels::removeChannel(const uint64_t& channelId) {
312 std::unique_lock<std::shared_mutex> theLock{ Channels::theMutex };
313 Channels::cache->erase(channelId);
314 };
