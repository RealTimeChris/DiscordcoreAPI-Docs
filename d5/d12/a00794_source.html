204  CoRoutine<Channel> Channels::getChannelAsync(GetChannelData dataPackage) {
205  DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Get_Channel };
206  co_await NewThreadAwaitable<Channel>();
207  workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Get;
208  workload.relativePath = "/channels/" + std::to_string(dataPackage.channelId);
209  workload.callStack = "Channels::getChannelAsync()";
210  Channel channel = Channels::httpsClient->submitWorkloadAndGetResult<Channel>(workload);
211  channel = Channels::getCachedChannelAsync(dataPackage).get();
212  co_return channel;
213  }

215  CoRoutine<ChannelData> Channels::getCachedChannelAsync(GetChannelData dataPackage) {
216  co_await NewThreadAwaitable<ChannelData>();
217  std::shared_lock theLock{ Channels::theMutex };
218  if (!Channels::cache->contains(dataPackage.channelId)) {
219  theLock.unlock();
220  auto theChannel = Channels::getChannelAsync(dataPackage).get();
221  Channels::insertChannel(theChannel);
222  co_return theChannel;
223  } else {
224  co_return *(*Channels::cache)[dataPackage.channelId];
225  }
226  }

378  void Channels::insertChannel(ChannelData channel) {
379  std::unique_lock theLock{ Channels::theMutex };
380  if (channel.id == 0) {
381  return;
382  }
383  auto newCache = std::make_unique<std::unordered_map<Snowflake, std::unique_ptr<ChannelData>>>();
384  for (auto& [key, value]: *Channels::cache) {
385  (*newCache)[key] = std::move(value);
386  }
387  if (Channels::configManager->doWeCacheChannels()) {
388  (*newCache)[channel.id] = std::make_unique<ChannelData>(channel);
389  }
390  Channels::cache.reset(nullptr);
391  Channels::cache = std::move(newCache);
392  }
