239 CreateInteractionResponseData::CreateInteractionResponseData(RespondToInputEventData& dataPackage) {
240 this->data = dataPackage;
241 if (dataPackage.eventType == InteractionType::Message_Component && dataPackage.type == InputEventResponseType::Deferred_Response) {
242 this->data.type = InteractionCallbackType::Deferred_Update_Message;
243 } else if (dataPackage.eventType == InteractionType::Message_Component) {
244 this->data.type = InteractionCallbackType::Update_Message;
245 } else if (dataPackage.eventType == InteractionType::Application_Command_Autocomplete ||
246 dataPackage.type == InputEventResponseType::Application_Command_AutoComplete_Result) {
247 this->data.type = InteractionCallbackType::Application_Command_Autocomplete_Result;
248 } else {
249 this->data.type = InteractionCallbackType::Channel_Message_With_Source;
250 }
251 if (dataPackage.type == InputEventResponseType::Modal_Interaction_Response || dataPackage.title != "") {
252 this->data.type = InteractionCallbackType::Modal;

254 this->interactionPackage.interactionToken = dataPackage.interactionToken;

415 CoRoutine<Message> Interactions::createInteractionResponseAsync(CreateInteractionResponseData dataPackage) {
416 DiscordCoreInternal::HttpsWorkloadData workload{ DiscordCoreInternal::HttpsWorkloadType::Post_Interaction_Response };
417 co_await NewThreadAwaitable<Message>();
418 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Post;
419 workload.relativePath =
420 "/interactions/" + std::to_string(dataPackage.interactionPackage.interactionId) + "/" + dataPackage.interactionPackage.interactionToken + "/callback";
421 if (dataPackage.data.data.files.size() > 0) {
422 workload.payloadType = DiscordCoreInternal::PayloadType::Multipart_Form;
423 workload.content = constructMultiPartData(nlohmann::json::parse(std::string{ dataPackage.data }), dataPackage.data.data.files);
424 } else {
425 workload.content = dataPackage.data;
426 }
427 workload.callStack = "Interactions::createInteractionResponseAsync()";
428 Interactions::httpsClient->submitWorkloadAndGetResult<void>(workload);
429 GetInteractionResponseData dataPackage01{};
430 dataPackage01.applicationId = dataPackage.interactionPackage.applicationId;
431 dataPackage01.interactionToken = dataPackage.interactionPackage.interactionToken;
432 if (dataPackage.data.type != InteractionCallbackType::Application_Command_Autocomplete_Result) {
433 co_return Interactions::getInteractionResponseAsync(dataPackage01).get();
434 } else {
435 co_return Message{};
436 }
437 }
