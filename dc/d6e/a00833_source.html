53 void WebSocketMessageHandler::createHeader(std::string& outBuffer, uint64_t sendLength, WebSocketOpCode opCode) noexcept {
54 try {
55 outBuffer.push_back(static_cast<uint8_t>(opCode) | webSocketFinishBit);
56
57 int32_t indexCount{ 0 };
58 if (sendLength <= webSocketMaxPayloadLengthSmall) {
59 outBuffer.push_back(static_cast<uint8_t>(sendLength));
60 indexCount = 0;
61 } else if (sendLength <= webSocketMaxPayloadLengthLarge) {
62 outBuffer.push_back(webSocketPayloadLengthMagicLarge);
63 indexCount = 2;
64 } else {
65 outBuffer.push_back(webSocketPayloadLengthMagicHuge);
66 indexCount = 8;
67 }
68 for (int32_t x = indexCount - 1; x >= 0; x--) {
69 outBuffer.push_back(static_cast<uint8_t>(sendLength >> x * 8));
70 }
71
72 outBuffer[1] |= webSocketMaskBit;
73 outBuffer.push_back(0);
74 outBuffer.push_back(0);
75 outBuffer.push_back(0);
76 outBuffer.push_back(0);
77 } catch (...) {
78 if (this->configManager->doWePrintWebSocketErrorMessages()) {
79 DiscordCoreAPI::reportException("WebSocketMessageHandler::createHeader()");
80 }
81 }

101 bool WebSocketMessageHandler::parseMessage(WebSocketSSLShard* theShard) noexcept {
102 if (theShard->inputBuffer.size() < 4) {
103 return true;
104 }
105 theShard->dataOpCode = static_cast<WebSocketOpCode>(theShard->inputBuffer[0] & ~webSocketFinishBit);
106 switch (theShard->dataOpCode) {
107 case WebSocketOpCode::Op_Continuation:
108 [[fallthrough]];
109 case WebSocketOpCode::Op_Text:
110 [[fallthrough]];
111 case WebSocketOpCode::Op_Binary:
112 [[fallthrough]];
113 case WebSocketOpCode::Op_Ping:
114 [[fallthrough]];
115 case WebSocketOpCode::Op_Pong: {
116 uint8_t length01 = theShard->inputBuffer[1];
117 theShard->messageOffset = 2;
118 if (length01 & webSocketMaskBit) {
119 return true;
120 }
121 theShard->messageLength = length01;
122 if (length01 == webSocketPayloadLengthMagicLarge) {
123 if (theShard->inputBuffer.size() < 8) {
124 return true;
125 }
126 uint8_t length03 = theShard->inputBuffer[2];
127 uint8_t length04 = theShard->inputBuffer[3];
128 theShard->messageLength = static_cast<uint64_t>((length03 << 8) | length04);
129 theShard->messageOffset += 2;
130 } else if (length01 == webSocketPayloadLengthMagicHuge) {
131 if (theShard->inputBuffer.size() < 10) {
132 return true;
133 }
134 theShard->messageLength = 0;
135 for (int64_t x = 2, shift = 56; x < 10; ++x, shift -= 8) {
136 uint8_t lengthNew = static_cast<uint8_t>(theShard->inputBuffer[x]);
137 theShard->messageLength |= static_cast<uint64_t>((lengthNew & static_cast<uint64_t>(0xff)) << static_cast<uint64_t>(shift));
138 }
139 theShard->messageOffset += 8;
140 }
141 if (theShard->inputBuffer.size() < static_cast<uint64_t>(theShard->messageOffset) + static_cast<uint64_t>(theShard->messageLength)) {
142 return true;
143 } else {
144 std::string finalMessage{};
145 finalMessage.insert(finalMessage.begin(), theShard->inputBuffer.begin() + theShard->messageOffset,
146 theShard->inputBuffer.begin() + theShard->messageOffset + theShard->messageLength);
147 theShard->processedMessages.push(finalMessage);
148 theShard->inputBuffer.erase(theShard->inputBuffer.begin(), theShard->inputBuffer.begin() + theShard->messageOffset + theShard->messageLength);
149 return true;
150 }
151 }
152 case WebSocketOpCode::Op_Close: {
153 uint16_t close = theShard->inputBuffer[2] & 0xff;
154 close <<= 8;
155 close |= theShard->inputBuffer[3] & 0xff;
156 theShard->closeCode = static_cast<WebSocketCloseCode>(close);
157 theShard->inputBuffer.erase(theShard->inputBuffer.begin(), theShard->inputBuffer.begin() + 4);
158 if (this->configManager->doWePrintWebSocketErrorMessages()) {
159 cout << DiscordCoreAPI::shiftToBrightRed() << "WebSocket " + theShard->shard.dump() + " Closed; Code: " << +static_cast<uint16_t>(theShard->closeCode)
160 << DiscordCoreAPI::reset() << endl
161 << endl;
162 }
163 return false;
164 }
165 default: {
166 return true;
167 }
168 }

235 bool WebSocketSSLShard::sendMessage(std::string& dataToSend, bool priority) noexcept {
236 if (this->theSSLState.load() == SSLConnectionState::Connected) {
237 try {
238 if (dataToSend.size() == 0) {
239 return false;
240 }
241 if (this->configManager->doWePrintWebSocketSuccessMessages()) {
242 cout << DiscordCoreAPI::shiftToBrightBlue() << "Sending WebSocket " + this->shard.dump() + std::string("'s Message: ") << endl
243 << dataToSend << DiscordCoreAPI::reset();
244 }
245 ProcessIOResult didWeWrite{ false };
246 DiscordCoreAPI::StopWatch theStopWatch{ 5000ms };
247 do {
248 if (theStopWatch.hasTimePassed()) {
249 this->onClosed();
250 return false;
251 }
252 std::this_thread::sleep_for(1ms);
253 didWeWrite = this->writeData(dataToSend, true);
254 } while (didWeWrite != ProcessIOResult::No_Error);
255 if (didWeWrite != ProcessIOResult::No_Error) {
256 this->onClosed();
257 return false;
258 }
259 return true;
260 } catch (...) {
261 if (this->configManager->doWePrintWebSocketErrorMessages()) {
262 DiscordCoreAPI::reportException("WebSocketSSLShard::sendMessage()");
263 }
264 this->onClosed();
265 return false;
266 }
267 }
