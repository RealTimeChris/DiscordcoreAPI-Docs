31 std::string VoiceConnection::encryptSingleAudioFrame(EncodedFrameData& bufferToSend, int32_t audioSSRC, const std::string& keys) {
32 if (keys.size() > 0) {
33 this->sequenceIndex++;
34 this->timeStamp += 960;
35 const int64_t nonceSize{ crypto_secretbox_NONCEBYTES };
36 const int64_t headerSize{ 12 };
37 const uint8_t byteSize{ 8 };
38 uint8_t headerFinal[headerSize]{};
39 headerFinal[0] = 0x80;
40 headerFinal[1] = 0x78;
41 headerFinal[2] = static_cast<uint8_t>(this->sequenceIndex >> (byteSize * 1));
42 headerFinal[3] = static_cast<uint8_t>(this->sequenceIndex >> (byteSize * 0));
43 headerFinal[4] = static_cast<uint8_t>(this->timeStamp >> (byteSize * 3));
44 headerFinal[5] = static_cast<uint8_t>(this->timeStamp >> (byteSize * 2));
45 headerFinal[6] = static_cast<uint8_t>(this->timeStamp >> (byteSize * 1));
46 headerFinal[7] = static_cast<uint8_t>(this->timeStamp >> (byteSize * 0));
47 headerFinal[8] = static_cast<uint8_t>(audioSSRC >> (byteSize * 3));
48 headerFinal[9] = static_cast<uint8_t>(audioSSRC >> (byteSize * 2));
49 headerFinal[10] = static_cast<uint8_t>(audioSSRC >> (byteSize * 1));
50 headerFinal[11] = static_cast<uint8_t>(audioSSRC >> (byteSize * 0));
51 uint8_t nonceForLibSodium[nonceSize]{};
52 for (uint64_t x = 0; x < headerSize; x++) {
53 nonceForLibSodium[x] = headerFinal[x];
54 }
55 for (int64_t x = headerSize; x < nonceSize; x++) {
56 nonceForLibSodium[x] = 0;
57 }
58 uint64_t numOfBytes{ headerSize + bufferToSend.data.size() + crypto_secretbox_MACBYTES };
59 std::unique_ptr<uint8_t[]> audioDataPacket{ std::make_unique<uint8_t[]>(numOfBytes) };
60 for (uint64_t x = 0; x < headerSize; x++) {
61 audioDataPacket[x] = headerFinal[x];
62 }
63 std::unique_ptr<uint8_t[]> encryptionKeys{ std::make_unique<uint8_t[]>(keys.size()) };
64 for (uint64_t x = 0; x < keys.size(); x++) {
65 encryptionKeys[x] = keys[x];
66 }
67 if (crypto_secretbox_easy(audioDataPacket.get() + headerSize, bufferToSend.data.data(), bufferToSend.data.size(), nonceForLibSodium, encryptionKeys.get()) != 0) {
68 return "";
69 };
70 bufferToSend.data.clear();
71 std::string audioDataPacketNew{};
72 audioDataPacketNew.insert(audioDataPacketNew.begin(), audioDataPacket.get(), audioDataPacket.get() + numOfBytes);
73 return audioDataPacketNew;
74 }
75 return std::string{};
76 }

153 bool VoiceConnection::connect(const DiscordCoreInternal::VoiceConnectInitData& voiceConnectInitDataNew) {
154 if (this->baseSocketAgent->theClients.contains(voiceConnectInitDataNew.currentShard) && this->baseSocketAgent->theClients[voiceConnectInitData.currentShard]) {
155 std::lock_guard<std::mutex> theLock{ this->baseSocketAgent->theClients[voiceConnectInitDataNew.currentShard]->accessorMutex };
156 this->voiceConnectInitData = voiceConnectInitDataNew;
157 this->areWeStopping.store(false);
158 this->stopSetEvent.set();
159 this->pauseEvent.set();
160 StopWatch theStopWatch{ 2000ms };
161 while (!this->baseSocketAgent->theClients[this->voiceConnectInitData.currentShard]->areWeConnected02.load()) {
162 std::this_thread::sleep_for(1ms);
163 if (theStopWatch.hasTimePassed()) {
164 return false;
165 }
166 }
167
168 if (!this->voiceSocketAgent) {
169 this->voiceSocketAgent = std::make_unique<DiscordCoreInternal::VoiceSocketAgent>(this->voiceConnectInitData, this->baseSocketAgent,

171 }
172 this->baseSocketAgent->getVoiceConnectionData(this->voiceConnectInitData, this->baseSocketAgent->theClients[this->voiceConnectInitData.currentShard].get());

174 this->doWeReconnect = &this->voiceSocketAgent->doWeReconnect;
175 this->doWeDisconnect = &this->voiceSocketAgent->doWeDisconnect;
176 theStopWatch.resetTimer();
177 while (!this->voiceSocketAgent->areWeConnected.load()) {
178 std::this_thread::sleep_for(1ms);
179 if (theStopWatch.hasTimePassed()) {
180 return false;
181 }
182 }
183 this->voiceConnectionData = this->voiceSocketAgent->voiceConnectionData;
184 if (!this->theTask) {
185 this->theTask = std::make_unique<std::jthread>([=, this](std::stop_token theToken) {
186 this->run(theToken);

188 }
189 theStopWatch.resetTimer();
190 while (!this->voiceSocketAgent->voiceSocket->areWeConnected.load()) {
191 std::this_thread::sleep_for(1ms);
192 if (theStopWatch.hasTimePassed()) {
193 return false;
194 }
195 }
196 this->sendSilence();
197 this->areWeConnectedBool = true;
198 return true;
199 }

271 void VoiceConnection::run(std::stop_token theToken) {
272 while (!theToken.stop_requested()) {
273 std::this_thread::sleep_for(1ms);
274 if (!this->didWeJustConnect) {
275 this->audioBuffer.clearContents();
276 this->clearAudioData();
277 this->didWeJustConnect = false;
278 }
279 if (!this->playSetEvent.wait(10000)) {
280 if (theToken.stop_requested()) {
281 this->areWePlaying.store(false);
282 return;
283 }
284 continue;
285 };
286 this->playSetEvent.reset();
287 start:
288 std::this_thread::sleep_for(1ms);
289 if (theToken.stop_requested()) {
290 this->areWePlaying.store(false);
291 return;
292 }
293 this->audioData.type = AudioFrameType::Unset;
294 this->audioData.encodedFrameData.data.clear();
295 this->audioData.rawFrameData.data.clear();
296 this->audioBuffer.tryReceive(this->audioData);
297 if ((this->audioData.encodedFrameData.sampleCount == 0 && this->audioData.rawFrameData.sampleCount == 0) || this->audioData.type == AudioFrameType::Unset) {
298 this->areWePlaying.store(false);
299 goto start;
300 }
301 this->areWePlaying.store(true);
302 int64_t startingValue{ std::chrono::duration_cast<std::chrono::nanoseconds, int64_t>(std::chrono::system_clock::now().time_since_epoch()).count() };
303 int64_t intervalCount{ 20000000 };
304 int32_t frameCounter{ 0 };
305 int64_t totalTime{ 0 };
306 if (this->disconnectStartTime != 0) {
307 int64_t currentTime = static_cast<int64_t>(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
308 if (currentTime - this->disconnectStartTime >= 60000) {
309 this->reconnect();
310 }
311 }
312 this->sendSpeakingMessage(true);
313 while (
314 (this->audioData.rawFrameData.sampleCount != 0 || this->audioData.encodedFrameData.sampleCount != 0) && !this->areWeStopping.load() && !theToken.stop_requested()) {
315 this->areWePlaying.store(true);
316 if (this->doWeReconnect->load()) {
317 this->areWeConnectedBool = false;
318 this->sendSpeakingMessage(false);
319 this->reconnect();
320 this->sendSpeakingMessage(true);
321 this->areWePlaying.store(true);
322 }
323 if (this->areWeStopping.load()) {
324 this->areWePlaying.store(false);
325 break;
326 }
327 this->pauseEvent.wait(240000);
328 if (theToken.stop_requested()) {
329 this->areWePlaying.store(false);
330 return;
331 }
332 frameCounter++;
333 this->audioBuffer.tryReceive(this->audioData);
334 if (this->audioData.guildMemberId != 0) {
335 this->currentGuildMemberId = this->audioData.guildMemberId;
336 }
337 nanoSleep(100000);
338 if (theToken.stop_requested()) {
339 this->areWePlaying.store(false);
340 return;
341 }
342 if (this->audioData.type != AudioFrameType::Unset && this->audioData.type != AudioFrameType::Skip && !this->areWeStopping.load()) {
343 std::string newFrame{};
344 if (this->audioData.type == AudioFrameType::RawPCM) {
345 std::vector<RawFrameData> rawFrames{};
346 rawFrames.push_back(this->audioData.rawFrameData);
347 auto encodedFrameData = this->encoder->encodeFrames(rawFrames);
348 newFrame = this->encryptSingleAudioFrame(encodedFrameData[0].encodedFrameData, this->voiceConnectionData.audioSSRC, this->voiceConnectionData.secretKey);
349 } else {
350 newFrame = this->encryptSingleAudioFrame(this->audioData.encodedFrameData, this->voiceConnectionData.audioSSRC, this->voiceConnectionData.secretKey);
351 }
352 if (newFrame.size() == 0) {
353 continue;
354 }
355 nanoSleep(18000000);
356 if (theToken.stop_requested()) {
357 this->areWePlaying.store(false);
358 return;
359 }
360 auto timeCounter =
361 static_cast<int64_t>(std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::system_clock::now().time_since_epoch()).count() - startingValue);
362 auto waitTime = intervalCount - timeCounter;
363 spinLock(waitTime);
364 startingValue = static_cast<int64_t>(std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
365 this->sendSingleAudioFrame(newFrame);
366 totalTime +=
367 static_cast<int64_t>(std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::system_clock::now().time_since_epoch()).count() - startingValue);
368 intervalCount = 20000000 - (totalTime / frameCounter);
369 this->audioData.type = AudioFrameType::Unset;
370 this->audioData.encodedFrameData.data.clear();
371 this->audioData.rawFrameData.data.clear();
372 } else if (this->audioData.type == AudioFrameType::Skip && !this->areWeStopping.load()) {
373 SongCompletionEventData completionEventData{};
374 completionEventData.guild = Guilds::getCachedGuildAsync({ .guildId = this->voiceConnectInitData.guildId }).get();
375 completionEventData.guildMember =
376 GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = this->currentGuildMemberId, .guildId = this->voiceConnectInitData.guildId }).get();
377 completionEventData.wasItAFail = false;
378 getSongAPIMap()[this->voiceConnectInitData.guildId]->onSongCompletionEvent(completionEventData);
379 break;
380 }
381 }
382 this->areWePlaying.store(false);
383 this->disconnectStartTime = static_cast<int64_t>(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
384 this->sendSpeakingMessage(false);
385 this->clearAudioData();
386 if (this->areWeStopping.load()) {
387 this->stopSetEvent.wait(5000);
388 this->stopSetEvent.reset();
389 this->areWeStopping.store(false);
390 }
391
392 if (theToken.stop_requested()) {
393 return;
394 }
395 }
