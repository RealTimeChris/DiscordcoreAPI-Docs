33  RTPPacket::operator std::string() {
34  if (this->theKeys.size() > 0) {
35  const uint8_t nonceSize{ crypto_secretbox_NONCEBYTES };
36  const uint8_t headerSize{ 12 };
37  const uint8_t byteSize{ 8 };
38  std::string header{};
39  storeBits(header, this->version);
40  storeBits(header, this->flags);
41  storeBits(header, this->sequence);
42  storeBits(header, this->timestamp);
43  storeBits(header, this->ssrc);
44  std::unique_ptr<uint8_t[]> nonceForLibSodium{ std::make_unique<uint8_t[]>(nonceSize) };
45  for (uint8_t x = 0; x < headerSize; x++) {
46  nonceForLibSodium[x] = header[x];
47  }
48  for (uint8_t x = headerSize; x < nonceSize; x++) {
49  nonceForLibSodium[x] = 0;
50  }
51  uint64_t numOfBytes{ headerSize + this->audioData.size() + crypto_secretbox_MACBYTES };
52  std::unique_ptr<uint8_t[]> audioDataPacket{ std::make_unique<uint8_t[]>(numOfBytes) };
53  for (uint8_t x = 0; x < headerSize; x++) {
54  audioDataPacket[x] = header[x];
55  }
56  std::unique_ptr<uint8_t[]> encryptionKeys{ std::make_unique<uint8_t[]>(this->theKeys.size()) };
57  for (uint64_t x = 0; x < this->theKeys.size(); x++) {
58  encryptionKeys[x] = this->theKeys[x];
59  }
60  if (crypto_secretbox_easy(audioDataPacket.get() + headerSize, this->audioData.data(), this->audioData.size(), nonceForLibSodium.get(), encryptionKeys.get()) != 0) {
61  return "";
62  };
63  std::string audioDataPacketNew{};
64  audioDataPacketNew.insert(audioDataPacketNew.begin(), audioDataPacket.get(), audioDataPacket.get() + numOfBytes);
65  return audioDataPacketNew;
66  }
67  return {};
68  }

114  void VoiceConnection::onMessageReceived() noexcept {

119  WebSocketSSLShard::processedMessages.pop();
120  } else {
121  return;
122  }
123  if (theMessage.size() > 0) {
124  nlohmann::json payload = payload.parse(theMessage);
125  if (this->configManager->doWePrintWebSocketSuccessMessages()) {
126  std::cout << DiscordCoreAPI::shiftToBrightGreen() << "Message received from Voice WebSocket: " << theMessage << DiscordCoreAPI::reset() << std::endl
127  << std::endl;
128  }
129  if (payload.contains("op") && !payload["op"].is_null()) {
130  switch (payload["op"].get<int32_t>()) {
131  case 2: {
132  this->voiceConnectionDataFinal.audioSSRC = payload["d"]["ssrc"].get<uint32_t>();
133  this->voiceConnectionDataFinal.voiceIp = payload["d"]["ip"].get<std::string>();
134  this->voiceConnectionDataFinal.voicePort = std::to_string(payload["d"]["port"].get<int64_t>());
135  for (auto& value: payload["d"]["modes"]) {
136  if (value == "xsalsa20_poly1305") {
137  this->voiceConnectionDataFinal.voiceEncryptionMode = value;
138  }
139  }
140  this->connectionState.store(VoiceConnectionState::Initializing_DatagramSocket);

142  }
143  case 4: {

244  this->lastActiveState.store(VoiceActiveState::Stopped);

248  this->activeState.store(VoiceActiveState::Connecting);

251  if (theStopWatch.hasTimePassed() || this->activeState.load() == VoiceActiveState::Exiting) {

264  }

266  this->sendHeartBeat();

268  }

270  DiscordCoreInternal::SSLClient::processIO(10000);

272  if (!stopToken.stop_requested() && WebSocketSSLShard::areWeStillConnected() && WebSocketSSLShard::inputBuffer.size() > 0) {
273  this->parseMessage(this);

276  this->onMessageReceived();
277  }
278  std::this_thread::sleep_for(1ms);
279  }
280  } catch (...) {
281  if (this->configManager->doWePrintWebSocketErrorMessages()) {
282  DiscordCoreAPI::reportException("VoiceConnection::run()");

285  }

312  void VoiceConnection::runVoice(std::stop_token stopToken) noexcept {
313  StopWatch theStopWatch{ 20000ms };
314  StopWatch theSendSilenceStopWatch{ 5000ms };
315  while (!stopToken.stop_requested() && !this->doWeQuit->load() && this->activeState.load() != VoiceActiveState::Exiting) {
316  if (!DatagramSocketClient::areWeStillConnected()) {
317  this->onClosed();
318  }
319  switch (this->activeState.load()) {
320  case VoiceActiveState::Connecting: {
321  while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Connecting) {
322  std::this_thread::sleep_for(1ms);
323  }
324  break;
325  }
326  case VoiceActiveState::Stopped: {
327  this->audioDataBuffer.clearContents();
328  this->clearAudioData();
329  while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Stopped) {
330  DatagramSocketClient::processIO(10000);
331  if (theSendSilenceStopWatch.hasTimePassed()) {
332  theSendSilenceStopWatch.resetTimer();
333  this->sendSpeakingMessage(true);
334  this->sendSpeakingMessage(false);
335  }
336  std::this_thread::sleep_for(1ms);
337  }
338  break;
339  }
340  case VoiceActiveState::Paused: {
341  while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Paused) {
342  DatagramSocketClient::processIO(10000);
343  if (theSendSilenceStopWatch.hasTimePassed()) {
344  theSendSilenceStopWatch.resetTimer();
345  this->sendSpeakingMessage(true);
346  this->sendSpeakingMessage(false);
347  }
348  std::this_thread::sleep_for(1ms);
349  }
350  break;
351  }
352  case VoiceActiveState::Playing: {
353  this->audioData.type = AudioFrameType::Unset;
354  this->audioData.encodedFrameData.data.clear();
355  this->audioData.rawFrameData.data.clear();
356  this->audioDataBuffer.tryReceive(this->audioData);
357  DoubleTimePointNs startingValue{ std::chrono::system_clock::now().time_since_epoch() };
358  DoubleTimePointNs intervalCount{ std::chrono::nanoseconds{ 20000000 } };
359  DoubleTimePointNs targetTime{ startingValue.time_since_epoch() + intervalCount.time_since_epoch() };
360  int32_t frameCounter{ 0 };
361  DoubleTimePointNs totalTime{ std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::nanoseconds{ 0 }) };
362  if (!this->areWePlaying.load()) {
363  this->sendSpeakingMessage(true);
364  this->areWePlaying.store(true);
365  }
366  while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Playing) {
367  theStopWatch.resetTimer();
368  while (!stopToken.stop_requested() && !DatagramSocketClient::areWeStillConnected()) {
369  if (theStopWatch.hasTimePassed() || this->activeState.load() == VoiceActiveState::Exiting) {
370  return;
371  }
372  std::this_thread::sleep_for(1ms);
373  }
374  frameCounter++;
375  if (this->audioData.guildMemberId != 0) {
376  this->currentGuildMemberId = this->audioData.guildMemberId;
377  }
378  if (this->audioData.type != AudioFrameType::Unset && this->audioData.type != AudioFrameType::Skip) {
379  std::string newFrame{};
380  if (this->audioData.type == AudioFrameType::RawPCM) {
381  std::vector<RawFrameData> rawFrames{};
382  rawFrames.push_back(this->audioData.rawFrameData);
383  auto encodedFrameData = this->encoder->encodeFrames(rawFrames);
384  newFrame = this->encryptSingleAudioFrame(encodedFrameData[0].encodedFrameData);
385  } else {
386  newFrame = this->encryptSingleAudioFrame(this->audioData.encodedFrameData);
387  }
388  if (newFrame.size() == 0) {
389  continue;
390  }
391  auto waitTime = targetTime - std::chrono::system_clock::now();
392  if (waitTime.count() > 500000 && !stopToken.stop_requested() && DatagramSocketClient::areWeStillConnected()) {
393  DatagramSocketClient::processIO(10000);
394  DatagramSocketClient::getInputBuffer();
395  }
396  waitTime = targetTime - std::chrono::system_clock::now();
397  nanoSleep(static_cast<int64_t>(static_cast<float>(waitTime.count()) * 0.95f));
398  waitTime = targetTime - std::chrono::system_clock::now();
399  if (waitTime.count() > 0) {
400  spinLock(waitTime.count());
401  }
402  startingValue = std::chrono::system_clock::now();
403  this->sendSingleAudioFrame(newFrame);
404  totalTime += std::chrono::system_clock::now() - startingValue;
405  auto intervalCountNew =
406  DoubleTimePointNs{ std::chrono::nanoseconds{ 20000000 } - totalTime.time_since_epoch() / frameCounter }.time_since_epoch().count();
407  intervalCount = DoubleTimePointNs{ std::chrono::nanoseconds{ static_cast<uint64_t>(intervalCountNew) } };
408  targetTime = std::chrono::system_clock::now().time_since_epoch() + intervalCount;
409  this->audioData.type = AudioFrameType::Unset;
410  this->audioData.encodedFrameData.data.clear();
411  this->audioData.rawFrameData.data.clear();
412  } else if (this->audioData.type == AudioFrameType::Skip) {
413  SongCompletionEventData completionEventData{};
414  completionEventData.guild = Guilds::getCachedGuildAsync({ .guildId = this->voiceConnectInitData.guildId }).get();
415  if (this->currentGuildMemberId != 0) {
416  completionEventData.guildMember =
417  GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = this->currentGuildMemberId, .guildId = this->voiceConnectInitData.guildId }).get();
418  }
419  completionEventData.wasItAFail = false;
420  getSongAPIMap()[this->voiceConnectInitData.guildId]->onSongCompletionEvent(completionEventData);
421  break;
422  } else {
423  break;
424  }
425  DatagramSocketClient::getInputBuffer();
426  this->audioDataBuffer.tryReceive(this->audioData);
427  }
428  DatagramSocketClient::getInputBuffer();
429  break;
430  }
431  case VoiceActiveState::Exiting: {
432  return;
433  }
434  default: {
435  break;
436  }
437  }
438  if (stopToken.stop_requested()) {
439  return;
440  }
441  }

482  void VoiceConnection::connectInternal() noexcept {
483  auto thePtr = this->baseSocketAgent->sslShards[voiceConnectInitData.currentShard].get();
484  std::unique_lock theLock{ thePtr->theMutex };
485  if (this->connections.size() > 0) {
486  this->connections.pop();
487  }
488  if (this->currentReconnectTries >= this->maxReconnectTries) {
489  this->doWeQuit->store(true);
490  if (this->configManager->doWePrintWebSocketErrorMessages()) {
491  std::cout << "VoiceConnection::connectInternal() Error: Failed to connect to voice channel!" << std::endl << std::endl;
492  }
493  }
494  switch (this->connectionState.load()) {
495  case VoiceConnectionState::Collecting_Init_Data: {
496  this->baseShard->voiceConnectionDataBufferMap[this->voiceConnectInitData.guildId] = &this->voiceConnectionDataBuffer;
497  this->baseShard->voiceConnectionDataBufferMap[this->voiceConnectInitData.guildId]->clearContents();
498  this->baseShard->getVoiceConnectionData(this->voiceConnectInitData);
499  if (DiscordCoreAPI::waitForTimeToPass(this->voiceConnectionDataBuffer, this->voiceConnectionDataFinal, 10000)) {
500  this->currentReconnectTries++;
501  this->onClosed();
502  this->connectInternal();
503  return;
504  }
505  this->baseUrl = this->voiceConnectionDataFinal.endPoint.substr(0, this->voiceConnectionDataFinal.endPoint.find(":"));
506  this->connectionState.store(VoiceConnectionState::Initializing_WebSocket);
507  this->connectInternal();
508  break;
509  }
510  case VoiceConnectionState::Initializing_WebSocket: {
511  if (!WebSocketSSLShard::connect(this->baseUrl, "443")) {
512  this->currentReconnectTries++;
513  this->onClosed();
514  this->connectInternal();
515  return;
516  }
517  this->theWebSocketState.store(DiscordCoreInternal::WebSocketSSLShardState::Upgrading);
518  std::string sendVector = "GET /?v=4 HTTP/1.1\r\nHost: " + this->baseUrl +
519  "\r\nPragma: no-cache\r\nUser-Agent: DiscordCoreAPI/1.0\r\nUpgrade: WebSocket\r\nConnection: Upgrade\r\nSec-WebSocket-Key: " +
520  DiscordCoreAPI::generateBase64EncodedKey() + "\r\nSec-WebSocket-Version: 13\r\n\r\n";
521  this->shard[0] = 0;
522  this->shard[1] = 1;
523  if (!this->sendMessage(sendVector, true)) {
524  this->currentReconnectTries++;
525  this->onClosed();
526  this->connectInternal();
527  return;
528  }
529  DiscordCoreInternal::SSLClient::processIO(200000);
530  if (!this->parseConnectionHeaders(this)) {
531  this->currentReconnectTries++;
532  this->onClosed();
533  this->connectInternal();
534  return;
535  }
536  this->connectionState.store(VoiceConnectionState::Collecting_Hello);
537  this->connectInternal();
538  break;
539  }
540  case VoiceConnectionState::Collecting_Hello: {
541  if (!this->collectAndProcessAMessage(VoiceConnectionState::Sending_Identify)) {
542  this->currentReconnectTries++;
543  this->onClosed();
544  this->connectInternal();
545  return;
546  }
547  this->currentReconnectTries = 0;
548  this->connectInternal();
549  break;
550  }
551  case VoiceConnectionState::Sending_Identify: {
552  this->haveWeReceivedHeartbeatAck = true;
553  DiscordCoreInternal::VoiceIdentifyData identifyData{};
554  identifyData.connectInitData = this->voiceConnectInitData;
555  identifyData.connectionData = this->voiceConnectionDataFinal;
556  std::string sendVector{};
557  this->stringifyJsonData(identifyData, sendVector, DiscordCoreInternal::WebSocketOpCode::Op_Text);
558  if (!this->sendMessage(sendVector, true)) {
559  this->currentReconnectTries++;
560  this->onClosed();
561  this->connectInternal();
