33 RTPPacket::operator std::string() {
34 if (this->theKeys.size() > 0) {
35 const uint8_t nonceSize{ crypto_secretbox_NONCEBYTES };
36 const uint8_t headerSize{ 12 };
37 const uint8_t byteSize{ 8 };
38 std::string header{};
39 storeBits(header, this->version);
40 storeBits(header, this->flags);
41 storeBits(header, this->sequence);
42 storeBits(header, this->timestamp);
43 storeBits(header, this->ssrc);
44 std::unique_ptr<uint8_t[]> nonceForLibSodium{ std::make_unique<uint8_t[]>(nonceSize) };
45 for (uint8_t x = 0; x < headerSize; x++) {
46 nonceForLibSodium[x] = header[x];
47 }
48 for (uint8_t x = headerSize; x < nonceSize; x++) {
49 nonceForLibSodium[x] = 0;
50 }
51 uint64_t numOfBytes{ headerSize + this->audioData.size() + crypto_secretbox_MACBYTES };
52 std::unique_ptr<uint8_t[]> audioDataPacket{ std::make_unique<uint8_t[]>(numOfBytes) };
53 for (uint8_t x = 0; x < headerSize; x++) {
54 audioDataPacket[x] = header[x];
55 }
56 std::unique_ptr<uint8_t[]> encryptionKeys{ std::make_unique<uint8_t[]>(this->theKeys.size()) };
57 for (uint64_t x = 0; x < this->theKeys.size(); x++) {
58 encryptionKeys[x] = this->theKeys[x];
59 }
60 if (crypto_secretbox_easy(audioDataPacket.get() + headerSize, this->audioData.data(), this->audioData.size(), nonceForLibSodium.get(), encryptionKeys.get()) != 0) {
61 return "";
62 };
63 std::string audioDataPacketNew{};
64 audioDataPacketNew.insert(audioDataPacketNew.begin(), audioDataPacket.get(), audioDataPacket.get() + numOfBytes);
65 return audioDataPacketNew;
66 }
67 return {};
68 }

114 void VoiceConnection::onMessageReceived() noexcept {
115 try {
116 std::string theMessage{};
117 if (WebSocketSSLShard::processedMessages.size() > 0) {
118 theMessage = WebSocketSSLShard::processedMessages.front();
119 WebSocketSSLShard::processedMessages.pop();
120 } else {
121 return;
122 }
123 if (theMessage.size() > 0) {
124 nlohmann::json payload = payload.parse(theMessage);
125 if (this->configManager->doWePrintWebSocketSuccessMessages()) {
126 std::cout << DiscordCoreAPI::shiftToBrightGreen() << "Message received from Voice WebSocket: " << theMessage << DiscordCoreAPI::reset() << std::endl
127 << std::endl;
128 }
129 if (payload.contains("op") && !payload["op"].is_null()) {
130 switch (payload["op"].get<int32_t>()) {
131 case 2: {
132 this->voiceConnectionDataFinal.audioSSRC = payload["d"]["ssrc"].get<uint32_t>();
133 this->voiceConnectionDataFinal.voiceIp = payload["d"]["ip"].get<std::string>();
134 this->voiceConnectionDataFinal.voicePort = std::to_string(payload["d"]["port"].get<int64_t>());
135 for (auto& value: payload["d"]["modes"]) {
136 if (value == "xsalsa20_poly1305") {
137 this->voiceConnectionDataFinal.voiceEncryptionMode = value;
138 }
139 }
140 this->connectionState.store(VoiceConnectionState::Initializing_DatagramSocket);
141 break;
142 }
143 case 4: {
144 for (uint32_t x = 0; x < payload["d"]["secret_key"].size(); x++) {
145 this->voiceConnectionDataFinal.secretKey.push_back(payload["d"]["secret_key"][x].get<uint8_t>());
146 }
147 this->connectionState.store(VoiceConnectionState::Collecting_Init_Data);
148 break;
149 }
150 case 6: {
151 this->haveWeReceivedHeartbeatAck = true;
152 break;
153 }
154 case 8: {
155 if (payload["d"].contains("heartbeat_interval")) {
156 this->heartbeatInterval = static_cast<int32_t>(payload["d"]["heartbeat_interval"].get<float>());
157 this->areWeHeartBeating = false;
158 this->connectionState.store(VoiceConnectionState::Sending_Identify);
159 }
160 break;
161 }
162 }
163 }
164 }
165 } catch (...) {
166 if (this->configManager->doWePrintWebSocketErrorMessages()) {
167 DiscordCoreAPI::reportException("VoiceConnection::onMessageReceived()");
168 }
169 this->onClosed();
170 }

238 void VoiceConnection::runWebSocket(std::stop_token& stopToken) noexcept {
239 try {
240 while (!stopToken.stop_requested() && !this->doWeQuit->load() && this->activeState.load() != VoiceActiveState::Exiting) {
241 if (!stopToken.stop_requested() && WebSocketSSLShard::connections->size() > 0) {
242 DiscordCoreAPI::StopWatch theStopWatch{ 10000ms };
243 if (this->activeState.load() == VoiceActiveState::Connecting) {
244 this->lastActiveState.store(VoiceActiveState::Stopped);
245 } else {
246 this->lastActiveState.store(this->activeState.load());
247 }
248 this->activeState.store(VoiceActiveState::Connecting);
249 this->connectionState.store(VoiceConnectionState::Collecting_Init_Data);
250 while (!stopToken.stop_requested() && this->baseShard->theWebSocketState.load() != DiscordCoreInternal::WebSocketSSLShardState::Authenticated) {
251 if (theStopWatch.hasTimePassed() || this->activeState.load() == VoiceActiveState::Exiting) {
252 return;
253 }
254 std::this_thread::sleep_for(1ms);
255 }
256 this->connectInternal();
257 this->sendSpeakingMessage(false);
258 this->sendSpeakingMessage(true);
259 this->activeState.store(this->lastActiveState.load());
260 }
261 if (!stopToken.stop_requested() && this->heartbeatInterval != 0 && !this->areWeHeartBeating) {
262 this->areWeHeartBeating = true;
263 this->heartBeatStopWatch = DiscordCoreAPI::StopWatch{ std::chrono::milliseconds{ this->heartbeatInterval } };
264 }
265 if (!stopToken.stop_requested() && WebSocketSSLShard::areWeStillConnected() && this->heartBeatStopWatch.hasTimePassed() && this->areWeHeartBeating) {
266 this->sendHeartBeat();
267 this->heartBeatStopWatch.resetTimer();
268 }
269 if (!stopToken.stop_requested() && WebSocketSSLShard::areWeStillConnected()) {
270 DiscordCoreInternal::SSLClient::processIO(10000);
271 }
272 if (!stopToken.stop_requested() && WebSocketSSLShard::areWeStillConnected() && WebSocketSSLShard::inputBuffer.size() > 0) {
273 this->parseMessage(this);
274 }
275 if (!stopToken.stop_requested() && WebSocketSSLShard::areWeStillConnected() && this->processedMessages.size() > 0) {
276 this->onMessageReceived();
277 }
278 std::this_thread::sleep_for(1ms);
279 }
280 } catch (...) {
281 if (this->configManager->doWePrintWebSocketErrorMessages()) {
282 DiscordCoreAPI::reportException("VoiceConnection::run()");
283 }
284 this->onClosed();
285 }

293 return false;
294 }

301 this->onMessageReceived();
302 return true;
303 }

305 return false;
306 }
307 std::this_thread::sleep_for(1ms);
308 }
309 return false;

312 void VoiceConnection::runVoice(std::stop_token& stopToken) noexcept {
313 StopWatch theStopWatch{ 20000ms };
314 StopWatch theSendSilenceStopWatch{ 5000ms };
315 while (!stopToken.stop_requested() && !this->doWeQuit->load() && this->activeState.load() != VoiceActiveState::Exiting) {
316 if (!DatagramSocketClient::areWeStillConnected()) {
317 this->onClosed();
318 }
319 switch (this->activeState.load()) {
320 case VoiceActiveState::Connecting: {
321 while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Connecting) {
322 std::this_thread::sleep_for(1ms);
323 }
324 break;
325 }
326 case VoiceActiveState::Stopped: {
327 this->areWePlaying.store(false);
328 this->audioDataBuffer.clearContents();
329 this->clearAudioData();
330 while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Stopped) {
331 DatagramSocketClient::processIO(10000);
332 if (theSendSilenceStopWatch.hasTimePassed()) {
333 theSendSilenceStopWatch.resetTimer();
334 this->sendSpeakingMessage(true);
335 this->sendSpeakingMessage(false);
336 }
337 std::this_thread::sleep_for(1ms);
338 }
339 break;
340 }
341 case VoiceActiveState::Paused: {
342 this->areWePlaying.store(false);
343 while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Paused) {
344 DatagramSocketClient::processIO(10000);
345 if (theSendSilenceStopWatch.hasTimePassed()) {
346 theSendSilenceStopWatch.resetTimer();
347 this->sendSpeakingMessage(true);
348 this->sendSpeakingMessage(false);
349 }
350 std::this_thread::sleep_for(1ms);
351 }
352 break;
353 }
354 case VoiceActiveState::Playing: {
355 this->audioData.type = AudioFrameType::Unset;
356 this->audioData.encodedFrameData.data.clear();
357 this->audioData.rawFrameData.data.clear();
358 this->audioDataBuffer.tryReceive(this->audioData);
359 DoubleTimePointNs startingValue{ std::chrono::system_clock::now().time_since_epoch() };
360 DoubleTimePointNs intervalCount{ std::chrono::nanoseconds{ 20000000 } };
361 DoubleTimePointNs targetTime{ startingValue.time_since_epoch() + intervalCount.time_since_epoch() };
362 int32_t frameCounter{ 0 };
363 DoubleTimePointNs totalTime{ std::chrono::duration_cast<std::chrono::nanoseconds>(std::chrono::nanoseconds{ 0 }) };
364 if (!this->areWePlaying.load()) {
365 this->sendSpeakingMessage(true);
366 this->areWePlaying.store(true);
367 }
368 while (!stopToken.stop_requested() && this->activeState.load() == VoiceActiveState::Playing) {
369 theStopWatch.resetTimer();
370 while (!stopToken.stop_requested() && !DatagramSocketClient::areWeStillConnected()) {
371 if (theStopWatch.hasTimePassed() || this->activeState.load() == VoiceActiveState::Exiting) {
372 return;
373 }
374 std::this_thread::sleep_for(1ms);
375 }
376 frameCounter++;
377 if (this->audioData.guildMemberId != 0) {
378 this->currentGuildMemberId = this->audioData.guildMemberId;
379 }
380 if (this->audioData.type != AudioFrameType::Unset && this->audioData.type != AudioFrameType::Skip) {
381 std::string newFrame{};
382 if (this->audioData.type == AudioFrameType::RawPCM) {
383 std::vector<RawFrameData> rawFrames{};
384 rawFrames.push_back(this->audioData.rawFrameData);
385 auto encodedFrameData = this->encoder.encodeFrames(rawFrames);
386 newFrame = this->encryptSingleAudioFrame(encodedFrameData[0].encodedFrameData);
387 } else {
388 newFrame = this->encryptSingleAudioFrame(this->audioData.encodedFrameData);
389 }
390 if (newFrame.size() == 0) {
391 continue;
392 }
393 auto waitTime = targetTime - std::chrono::system_clock::now();
394 if (waitTime.count() > 500000 && !stopToken.stop_requested() && DatagramSocketClient::areWeStillConnected()) {
395 DatagramSocketClient::processIO(10000);
396 DatagramSocketClient::getInputBuffer();
397 }
398 waitTime = targetTime - std::chrono::system_clock::now();
399 nanoSleep(static_cast<int64_t>(static_cast<float>(waitTime.count()) * 0.95f));
400 waitTime = targetTime - std::chrono::system_clock::now();
401 if (waitTime.count() > 0) {
402 spinLock(waitTime.count());
403 }
404 startingValue = std::chrono::system_clock::now();
405 this->sendSingleAudioFrame(newFrame);
406 totalTime += std::chrono::system_clock::now() - startingValue;
407 auto intervalCountNew =
408 DoubleTimePointNs{ std::chrono::nanoseconds{ 20000000 } - totalTime.time_since_epoch() / frameCounter }.time_since_epoch().count();
409 intervalCount = DoubleTimePointNs{ std::chrono::nanoseconds{ static_cast<uint64_t>(intervalCountNew) } };
410 targetTime = std::chrono::system_clock::now().time_since_epoch() + intervalCount;
411 this->audioData.type = AudioFrameType::Unset;
412 this->audioData.encodedFrameData.data.clear();
413 this->audioData.rawFrameData.data.clear();
414 } else if (this->audioData.type == AudioFrameType::Skip) {
415 SongCompletionEventData completionEventData{};
416 completionEventData.guild = Guilds::getCachedGuildAsync({ .guildId = this->voiceConnectInitData.guildId }).get();
417 if (this->currentGuildMemberId != 0) {
418 completionEventData.guildMember =
419 GuildMembers::getCachedGuildMemberAsync({ .guildMemberId = this->currentGuildMemberId, .guildId = this->voiceConnectInitData.guildId }).get();
420 }
421 completionEventData.wasItAFail = false;
422 getSongAPIMap()[this->voiceConnectInitData.guildId]->onSongCompletionEvent(completionEventData);
423 break;
424 } else {
425 break;
426 }
427 DatagramSocketClient::getInputBuffer();
428 this->audioDataBuffer.tryReceive(this->audioData);
429 }
430 DatagramSocketClient::getInputBuffer();
431 break;
432 }
433 case VoiceActiveState::Exiting: {
434 return;
435 }
436 default: {
437 break;
438 }
439 }
440 if (stopToken.stop_requested()) {
441 return;
442 }
443 }

484 void VoiceConnection::connectInternal() noexcept {
485 auto thePtr = this->baseSocketAgent->sslShards[voiceConnectInitData.currentShard].get();
486 std::unique_lock theLock{ thePtr->theMutex };
487 if (this->connections.size() > 0) {
488 this->connections.pop();
489 }
490 if (this->currentReconnectTries >= this->maxReconnectTries) {
491 this->doWeQuit->store(true);
492 if (this->configManager->doWePrintWebSocketErrorMessages()) {
493 std::cout << "VoiceConnection::connectInternal() Error: Failed to connect to voice channel!" << std::endl << std::endl;
494 }
495 }
496 switch (this->connectionState.load()) {
497 case VoiceConnectionState::Collecting_Init_Data: {
498 this->baseShard->voiceConnectionDataBufferMap[this->voiceConnectInitData.guildId] = &this->voiceConnectionDataBuffer;
499 this->baseShard->voiceConnectionDataBufferMap[this->voiceConnectInitData.guildId]->clearContents();
500 this->baseShard->getVoiceConnectionData(this->voiceConnectInitData);
501 if (DiscordCoreAPI::waitForTimeToPass(this->voiceConnectionDataBuffer, this->voiceConnectionDataFinal, 10000)) {
502 this->currentReconnectTries++;
503 this->onClosed();
504 this->connectInternal();
505 return;
506 }
507 this->baseUrl = this->voiceConnectionDataFinal.endPoint.substr(0, this->voiceConnectionDataFinal.endPoint.find(":"));
508 this->connectionState.store(VoiceConnectionState::Initializing_WebSocket);
509 this->connectInternal();
510 break;
511 }
512 case VoiceConnectionState::Initializing_WebSocket: {
513 if (!WebSocketSSLShard::connect(this->baseUrl, "443")) {
514 this->currentReconnectTries++;
515 this->onClosed();
516 this->connectInternal();
517 return;
518 }
519 this->theWebSocketState.store(DiscordCoreInternal::WebSocketSSLShardState::Upgrading);
520 std::string sendVector = "GET /?v=4 HTTP/1.1\r\nHost: " + this->baseUrl +
521 "\r\nPragma: no-cache\r\nUser-Agent: DiscordCoreAPI/1.0\r\nUpgrade: WebSocket\r\nConnection: Upgrade\r\nSec-WebSocket-Key: " +
522 DiscordCoreAPI::generateBase64EncodedKey() + "\r\nSec-WebSocket-Version: 13\r\n\r\n";
523 this->shard[0] = 0;
524 this->shard[1] = 1;
525 if (!this->sendMessage(sendVector, true)) {
526 this->currentReconnectTries++;
527 this->onClosed();
528 this->connectInternal();
529 return;
530 }
531 DiscordCoreInternal::SSLClient::processIO(200000);
532 if (!this->parseConnectionHeaders(this)) {
533 this->currentReconnectTries++;
534 this->onClosed();
535 this->connectInternal();
536 return;
537 }
538 this->connectionState.store(VoiceConnectionState::Collecting_Hello);
539 this->connectInternal();
540 break;
541 }
542 case VoiceConnectionState::Collecting_Hello: {
543 if (!this->collectAndProcessAMessage(VoiceConnectionState::Sending_Identify)) {
544 this->currentReconnectTries++;
545 this->onClosed();
546 this->connectInternal();
547 return;
548 }
549 this->currentReconnectTries = 0;
550 this->connectInternal();
551 break;
552 }
553 case VoiceConnectionState::Sending_Identify: {
554 this->haveWeReceivedHeartbeatAck = true;
555 DiscordCoreInternal::VoiceIdentifyData identifyData{};
556 identifyData.connectInitData = this->voiceConnectInitData;
557 identifyData.connectionData = this->voiceConnectionDataFinal;
558 std::string sendVector{};
559 this->stringifyJsonData(identifyData, sendVector, DiscordCoreInternal::WebSocketOpCode::Op_Text);
560 if (!this->sendMessage(sendVector, true)) {
561 this->currentReconnectTries++;
562 this->onClosed();
563 this->connectInternal();
564 return;
565 }
566 this->connectionState.store(VoiceConnectionState::Collecting_Ready);
567 this->connectInternal();
568 break;
569 }
570 case VoiceConnectionState::Collecting_Ready: {
571 if (!this->collectAndProcessAMessage(VoiceConnectionState::Initializing_DatagramSocket)) {
572 this->currentReconnectTries++;
573 this->onClosed();
574 this->connectInternal();
575 return;
576 }
577 this->connectInternal();
578 break;
579 }
580 case VoiceConnectionState::Initializing_DatagramSocket: {
581 if (!this->voiceConnect()) {
582 this->currentReconnectTries++;
583 this->onClosed();
584 this->connectInternal();
585 return;
586 }
587 this->connectionState.store(VoiceConnectionState::Sending_Select_Protocol);
588 this->connectInternal();
589 break;
590 }
591 case VoiceConnectionState::Sending_Select_Protocol: {
592 DiscordCoreInternal::VoiceSocketProtocolPayloadData protocolPayloadData{};
593 protocolPayloadData.externalIp = this->voiceConnectionDataFinal.externalIp;
594 protocolPayloadData.voiceEncryptionMode = this->voiceConnectionDataFinal.voiceEncryptionMode;
595 protocolPayloadData.voicePort = this->voiceConnectionDataFinal.voicePort;
596 nlohmann::json protocolPayloadSelectString = protocolPayloadData;
597 std::string sendVector{};
598 this->stringifyJsonData(protocolPayloadSelectString, sendVector, DiscordCoreInternal::WebSocketOpCode::Op_Text);
599 if (!this->sendMessage(sendVector, true)) {
600 this->currentReconnectTries++;
601 this->onClosed();
602 this->connectInternal();
603 return;
604 }
605 this->connectionState.store(VoiceConnectionState::Collecting_Session_Description);
606 this->connectInternal();
607 break;
