386  std::string TimeStamp::getFutureISO8601TimeStamp(int32_t minutesToAdd, int32_t hoursToAdd, int32_t daysToAdd, int32_t monthsToAdd, int32_t yearsToAdd) {
387  std::time_t result = std::time(nullptr);
388  int32_t secondsPerMinute{ 60 };
389  int32_t minutesPerHour{ 60 };
390  int32_t secondsPerHour{ minutesPerHour * secondsPerMinute };
391  int32_t hoursPerDay{ 24 };
392  int32_t secondsPerDay{ secondsPerHour * hoursPerDay };
393  int32_t daysPerMonth{ 30 };
394  int32_t secondsPerMonth{ secondsPerDay * daysPerMonth };
395  int32_t daysPerYear{ 365 };
396  int32_t secondsPerYear{ secondsPerDay * daysPerYear };
397  int32_t secondsToAdd =
398  (yearsToAdd * secondsPerYear) + (monthsToAdd * secondsPerMonth) + (daysToAdd * secondsPerDay) + (hoursToAdd * secondsPerHour) + (minutesToAdd * secondsPerMinute);
399  result += secondsToAdd;
400  auto resultTwo = std::localtime(&result);
401  std::string theReturnString{};
402  if (resultTwo->tm_isdst) {
403  if (resultTwo->tm_hour + 4 >= 24) {
404  resultTwo->tm_hour = resultTwo->tm_hour - 24;
405  resultTwo->tm_mday++;
406  }
407  theReturnString = TimeStamp::getISO8601TimeStamp(std::to_string(resultTwo->tm_year + 1900), std::to_string(resultTwo->tm_mon + 1), std::to_string(resultTwo->tm_mday),
408  std::to_string(resultTwo->tm_hour + 4), std::to_string(resultTwo->tm_min), std::to_string(resultTwo->tm_sec));
409  } else {
410  if (resultTwo->tm_hour + 5 >= 24) {
411  resultTwo->tm_hour = resultTwo->tm_hour - 24;
412  resultTwo->tm_mday++;
413  }
414  theReturnString = TimeStamp::getISO8601TimeStamp(std::to_string(resultTwo->tm_year + 1900), std::to_string(resultTwo->tm_mon + 1), std::to_string(resultTwo->tm_mday),
415  std::to_string(resultTwo->tm_hour + 5), std::to_string(resultTwo->tm_min), std::to_string(resultTwo->tm_sec));
416  }
417  return theReturnString;
418  }

435  std::string TimeStamp::convertMsToDurationString(uint64_t durationInMs) {
436  std::string newString{};
437  uint64_t msPerSecond{ 1000 };
438  uint64_t secondsPerMinute{ 60 };
439  uint64_t minutesPerHour{ 60 };
440  uint64_t msPerMinute{ msPerSecond * secondsPerMinute };
441  uint64_t msPerHour{ msPerMinute * minutesPerHour };
442  uint64_t hoursLeft = static_cast<uint64_t>(trunc(durationInMs / msPerHour));
443  uint64_t minutesLeft = static_cast<uint64_t>(trunc((durationInMs % msPerHour) / msPerMinute));
444  uint64_t secondsLeft = static_cast<uint64_t>(trunc(((durationInMs % msPerHour) % msPerMinute) / msPerSecond));
445  if (hoursLeft >= 1) {
446  newString += std::to_string(hoursLeft) + " Hours, ";
447  newString += std::to_string(minutesLeft) + " Minutes, ";
448  newString += std::to_string(secondsLeft) + " Seconds.";
449  } else if (minutesLeft >= 1) {
450  newString += std::to_string(minutesLeft) + " Minutes, ";
451  newString += std::to_string(secondsLeft) + " Seconds.";
452  } else {
453  newString += std::to_string(secondsLeft) + " Seconds.";
454  }
