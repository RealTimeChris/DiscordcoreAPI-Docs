34 /// For updating a User's presence. \brief For updating a User's presence.
35 struct DiscordCoreAPI_Dll UpdatePresenceData {
36 std::vector<ActivityData> activities{};///< A std::vector of activities.
37 std::string status{};///< Current status.
38 int64_t since{ 0 };///< When was the activity started?
39 bool afk{ false };///< Are we afk.
40
41 operator nlohmann::json() {
42 nlohmann::json data{};
43 data["d"]["activities"] = nlohmann::json{};
44 for (auto& value: this->activities) {
45 nlohmann::json dataNew{};
46 if (value.url != "") {
47 dataNew["url"] = value.url;
48 }
49 dataNew["name"] = value.name;
50 dataNew["type"] = value.type;
51 data["d"]["activities"].push_back(dataNew);
52 }
53 data["d"]["status"] = this->status;
54 data["d"]["since"] = this->since;
55 data["d"]["afk"] = this->afk;
56 data["op"] = 3;
57 return data;
58 }
59 };
