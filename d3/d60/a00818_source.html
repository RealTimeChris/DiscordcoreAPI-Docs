33 VoiceConnection* GuildData::connectToVoice(const Snowflake guildMemberId, const Snowflake channelId, bool selfDeaf, bool selfMute) {
34 if (getVoiceConnectionMap().contains(this->id) && getVoiceConnectionMap()[this->id] && getVoiceConnectionMap()[this->id]->areWeConnected()) {
35 this->voiceConnectionPtr = getVoiceConnectionMap()[this->id].get();
36 return this->voiceConnectionPtr;
37 } else if (guildMemberId != 0 || channelId != 0) {
38 Snowflake theChannelId{};
39 if (guildMemberId != 0) {
40 if (this->voiceStates.contains(guildMemberId)) {
41 theChannelId = this->voiceStates[guildMemberId].channelId;
42 }
43 } else {
44 theChannelId = channelId;
45 }
46 std::string theShardId{ std::to_string((this->id >> 22) % this->discordCoreClient->configManager.getTotalShardCount()) };
47 auto theBaseSocketAgentIndex{ static_cast<int32_t>(
48 floor(static_cast<float>(stod(theShardId)) / static_cast<float>(this->discordCoreClient->configManager.getTotalShardCount())) *
49 static_cast<float>(std::thread::hardware_concurrency())) };
50 getVoiceConnectionMap()[this->id] = std::make_unique<VoiceConnection>(this->discordCoreClient->baseSocketAgentMap[std::to_string(theBaseSocketAgentIndex)].get());
51 this->voiceConnectionPtr = getVoiceConnectionMap()[this->id].get();
52 DiscordCoreInternal::VoiceConnectInitData voiceConnectInitData{};
53 voiceConnectInitData.currentShard = stoll(theShardId);
54 voiceConnectInitData.channelId = theChannelId;
55 voiceConnectInitData.guildId = this->id;
56 voiceConnectInitData.userId = this->discordCoreClient->getBotUser().id;
57 voiceConnectInitData.selfDeaf = selfDeaf;
58 voiceConnectInitData.selfMute = selfMute;
59 this->voiceConnectionPtr->connect(voiceConnectInitData);
60 return this->voiceConnectionPtr;
61 } else {
62 return nullptr;
63 }
64 }

70 void GuildData::disconnect() {
71 if (getVoiceConnectionMap().contains(this->id) && getVoiceConnectionMap()[this->id]) {
72 UpdateVoiceStateData updateVoiceData{};
73 updateVoiceData.channelId = 0;
74 updateVoiceData.selfDeaf = false;
75 updateVoiceData.selfMute = false;
76 updateVoiceData.guildId = this->id;
77 this->discordCoreClient->getBotUser().updateVoiceStatus(updateVoiceData);
78 SongAPI::stop(this->id);
79 getVoiceConnectionMap()[this->id]->disconnect();
80 getVoiceConnectionMap()[this->id].reset(nullptr);
81 this->voiceConnectionPtr = nullptr;
82 }

97 void GuildData::initialize() {
98 if (!getVoiceConnectionMap().contains(this->id)) {
99 std::string theShardId{ std::to_string((this->id >> 22) % this->discordCoreClient->configManager.getTotalShardCount()) };
100 getVoiceConnectionMap()[this->id] = std::make_unique<VoiceConnection>(this->discordCoreClient->baseSocketAgentMap[theShardId].get());
101 }
102 this->voiceConnectionPtr = getVoiceConnectionMap()[this->id].get();
103 if (!getYouTubeAPIMap().contains(this->id)) {
104 getYouTubeAPIMap()[this->id] =
105 std::make_unique<DiscordCoreInternal::YouTubeAPI>(this->id, this->discordCoreClient->httpsClient.get(), &this->discordCoreClient->configManager);
106 }
107 if (!getSoundCloudAPIMap().contains(this->id)) {
108 getSoundCloudAPIMap()[this->id] =
109 std::make_unique<DiscordCoreInternal::SoundCloudAPI>(this->id, this->discordCoreClient->httpsClient.get(), &this->discordCoreClient->configManager);
110 }
111 if (!getSongAPIMap().contains(this->id)) {
112 getSongAPIMap()[this->id] = std::make_unique<SongAPI>(this->id);
113 }
114 }
