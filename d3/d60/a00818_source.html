107 VoiceConnection* GuildData::connectToVoice(const Snowflake guildMemberId, const Snowflake channelId, bool selfDeaf, bool selfMute) {
108 if (getVoiceConnectionMap().contains(this->id) && getVoiceConnectionMap()[this->id] && getVoiceConnectionMap()[this->id]->areWeConnected()) {
109 this->voiceConnectionPtr = getVoiceConnectionMap()[this->id].get();
110 return this->voiceConnectionPtr;
111 } else if (guildMemberId != 0 || channelId != 0) {
112 Snowflake theChannelId{};
113 if (guildMemberId != 0) {
114 if (this->voiceStates.contains(guildMemberId)) {
115 theChannelId = this->voiceStates[guildMemberId].channelId;
116 }
117 } else {
118 theChannelId = channelId;
119 }
120 uint64_t theShardId{ (this->id >> 22) % this->discordCoreClient->configManager.getTotalShardCount() };
121 auto theBaseSocketAgentIndex{ static_cast<int32_t>(
122 floor(static_cast<float>(theShardId) / static_cast<float>(this->discordCoreClient->configManager.getTotalShardCount())) *
123 static_cast<float>(std::thread::hardware_concurrency())) };
124 DiscordCoreInternal::VoiceConnectInitData voiceConnectInitData{};
125 voiceConnectInitData.currentShard = theShardId;
126 voiceConnectInitData.channelId = theChannelId;
127 voiceConnectInitData.guildId = this->id;
128 voiceConnectInitData.userId = this->discordCoreClient->getBotUser().id;
129 voiceConnectInitData.selfDeaf = selfDeaf;
130 voiceConnectInitData.selfMute = selfMute;
131 StopWatch theStopWatch{ 10000ms };
132 this->discordCoreClient->baseSocketAgentMap[std::to_string(theBaseSocketAgentIndex)]->connectVoiceChannel(voiceConnectInitData);
133 while (!getVoiceConnectionMap()[this->id]->areWeConnected()) {
134 std::this_thread::sleep_for(1ms);
135 if (theStopWatch.hasTimePassed()) {
136 break;
137 }
138 }
139 this->voiceConnectionPtr = getVoiceConnectionMap()[this->id].get();
140 return this->voiceConnectionPtr;
141 } else {
142 return nullptr;
143 }
144 }

150 void GuildData::disconnect() {
151 if (getVoiceConnectionMap().contains(this->id) && getVoiceConnectionMap()[this->id].get()) {
152 UpdateVoiceStateData updateVoiceData{};
153 updateVoiceData.channelId = 0;
154 updateVoiceData.selfDeaf = false;
155 updateVoiceData.selfMute = false;
156 updateVoiceData.guildId = this->id;
157 this->discordCoreClient->getBotUser().updateVoiceStatus(updateVoiceData);
158 getVoiceConnectionMap()[this->id]->disconnect();
159 StopWatch theStopWatch{ 10000ms };
160 while (getVoiceConnectionMap()[this->id]->areWeConnectedBool.load()) {
161 std::this_thread::sleep_for(1ms);
162 if (theStopWatch.hasTimePassed()) {
163 break;
164 }
165 }
166 this->voiceConnectionPtr = nullptr;
167 }

184 std::string theShardId{ std::to_string((this->id >> 22) % this->discordCoreClient->configManager.getTotalShardCount()) };
