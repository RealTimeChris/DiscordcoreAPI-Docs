169  CoRoutine<MessageCollectorReturnData> MessageCollector::collectMessages(int32_t quantityToCollect, int32_t msToCollectForNew, MessageFilter filteringFunctionNew) {
170  co_await NewThreadAwaitable<MessageCollectorReturnData>();
171  this->quantityOfMessageToCollect = quantityToCollect;
172  this->filteringFunction = filteringFunctionNew;
173  this->msToCollectFor = msToCollectForNew;
174  this->collectorId = std::to_string(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
175  MessageCollector::messagesBufferMap[this->collectorId] = &this->messagesBuffer;
176  this->run();
177  co_return this->messageReturnData;
178  }
179 
180  void MessageCollector::run() {
181  int64_t startingTime = static_cast<int64_t>(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
182  int64_t elapsedTime{ 0 };
183  while (elapsedTime < this->msToCollectFor) {
184  Message message{};
185  waitForTimeToPass<Message>(this->messagesBuffer, message, static_cast<int32_t>(this->msToCollectFor - elapsedTime));
186  if (this->filteringFunction(message)) {
187  this->messageReturnData.messages.push_back(message);
188  }
189  if (static_cast<int32_t>(this->messageReturnData.messages.size()) >= this->quantityOfMessageToCollect) {
190  break;
191  }
192 
193  elapsedTime = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count() - startingTime;
194  }
195  }

295  CoRoutine<void> Messages::deleteMessageAsync(DeleteMessageData dataPackage) {
296  std::unique_ptr<DiscordCoreInternal::HttpsWorkloadData> workload{ nullptr };
297  bool hasTimeElapsedNew = dataPackage.timeStamp.hasTimeElapsed(14);
298  if (hasTimeElapsedNew) {
299  workload = std::make_unique<DiscordCoreInternal::HttpsWorkloadData>(DiscordCoreInternal::HttpsWorkloadType::Delete_Message_Old);
300  } else {
301  workload = std::make_unique<DiscordCoreInternal::HttpsWorkloadData>(DiscordCoreInternal::HttpsWorkloadType::Delete_Message);
302  }
303  co_await NewThreadAwaitable<void>();
304  std::this_thread::sleep_for(std::chrono::milliseconds{ dataPackage.timeDelay });
305  workload->workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Delete;
306  workload->relativePath = "/channels/" + std::to_string(dataPackage.channelId) + "/messages/" + std::to_string(dataPackage.messageId);
307  workload->callStack = "Messages::deleteMessageAsync()";
308  if (dataPackage.reason != "") {
309  workload->headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
310  }
311  co_return Messages::httpsClient->submitWorkloadAndGetResult<void>(*workload);
312  }
