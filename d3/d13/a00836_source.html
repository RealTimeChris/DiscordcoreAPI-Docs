85 void BotUser::updateVoiceStatus(UpdateVoiceStateData& dataPackage) {
86 nlohmann::json payload = dataPackage;
87 std::string theString{};
88 if (this->baseSocketAgent) {
89 int32_t theIndex{};
90 for (auto& [key, value]: this->baseSocketAgent->sslShards) {
91 theIndex = key;
92 break;
93 }
94 std::string theString{};
95 if (static_cast<DiscordCoreInternal::WebSocketSSLShard*>(this->baseSocketAgent->sslShards[theIndex].get())->dataOpCode ==
96 DiscordCoreInternal::WebSocketOpCode::Op_Binary) {
97 this->baseSocketAgent->sslShards[theIndex]->stringifyJsonData(payload, theString, DiscordCoreInternal::WebSocketOpCode::Op_Binary);
98 } else {
99 this->baseSocketAgent->sslShards[theIndex]->stringifyJsonData(payload, theString, DiscordCoreInternal::WebSocketOpCode::Op_Text);
100 }
101 this->baseSocketAgent->sslShards[theIndex]->sendMessage(theString, true);
102 }

179 CoRoutine<UserData> Users::getCachedUserAsync(GetUserData dataPackage) {
180 co_await NewThreadAwaitable<UserData>();
181 std::shared_lock theLock{ Users::theMutex };
182 if (!Users::cache->contains(dataPackage.userId)) {
183 theLock.unlock();
184 auto theUser = getUserAsync(dataPackage).get();
185 Users::insertUser(theUser);
186 co_return theUser;
187
188 } else {
189 co_return *(*Users::cache)[dataPackage.userId];
190 }
191 }

247 void Users::insertUser(UserData user) {
248 std::unique_lock theLock{ Users::theMutex };
249 if (user.id == 0) {
250 return;
251 }
252 auto newCache = std::make_unique<std::unordered_map<Snowflake, std::unique_ptr<UserData>>>();
253 for (auto& [key, value]: *Users::cache) {
254 (*newCache)[key] = std::move(value);
255 }
256 if (Users::configManager->doWeCacheUsers()) {
257 (*newCache)[user.id] = std::make_unique<UserData>(user);
258 }
259 Users::cache.reset(nullptr);
260 Users::cache = std::move(newCache);
261 }
