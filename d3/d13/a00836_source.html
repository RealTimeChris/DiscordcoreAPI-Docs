49 CoRoutine<MessageCollectorReturnData> MessageCollector::collectMessages(int32_t quantityToCollect, int32_t msToCollectForNew, MessageFilter filteringFunctionNew) {
50 co_await NewThreadAwaitable<MessageCollectorReturnData>();
51 this->quantityOfMessageToCollect = quantityToCollect;
52 this->filteringFunction = filteringFunctionNew;
53 this->msToCollectFor = msToCollectForNew;
54 this->collectorId = std::to_string(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
55 MessageCollector::messagesBufferMap[this->collectorId] = &this->messagesBuffer;
56 this->run();
57 co_return this->messageReturnData;
58 }
59
60 void MessageCollector::run() {
61 int64_t startingTime = static_cast<int64_t>(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
62 int64_t elapsedTime{ 0 };
63 while (elapsedTime < this->msToCollectFor) {
64 Message message{};
65 waitForTimeToPass<Message>(this->messagesBuffer, message, static_cast<int32_t>(this->msToCollectFor - elapsedTime));
66 if (this->filteringFunction(message)) {
67 this->messageReturnData.messages.push_back(message);
68 }
69 if (static_cast<int32_t>(this->messageReturnData.messages.size()) >= this->quantityOfMessageToCollect) {
70 break;
71 }
72
73 elapsedTime = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count() - startingTime;
74 }
75 }

78 if (MessageCollector::messagesBufferMap.contains(this->collectorId)) {
79 MessageCollector::messagesBufferMap.erase(this->collectorId);
80 }

185 CoRoutine<void> Messages::deleteMessageAsync(DeleteMessageData dataPackage) {
186 DiscordCoreInternal::HttpsWorkloadData workload{};
187 bool hasTimeElapsedNew = dataPackage.timeStamp.hasTimeElapsed(14);
188 if (hasTimeElapsedNew) {
189 workload.thisWorkerId = DiscordCoreInternal::HttpsWorkloadData::getAndIncrementWorkloadId(DiscordCoreInternal::HttpsWorkloadType::Delete_Message_Old);
190 workload.workloadType = DiscordCoreInternal::HttpsWorkloadType::Delete_Message_Old;
191 } else {
192 workload.thisWorkerId = DiscordCoreInternal::HttpsWorkloadData::getAndIncrementWorkloadId(DiscordCoreInternal::HttpsWorkloadType::Delete_Message);
193 workload.workloadType = DiscordCoreInternal::HttpsWorkloadType::Delete_Message;
194 }
195 co_await NewThreadAwaitable<void>();
196 std::this_thread::sleep_for(std::chrono::milliseconds{ dataPackage.timeDelay });
197 workload.workloadClass = DiscordCoreInternal::HttpsWorkloadClass::Delete;
198 workload.relativePath = "/channels/" + std::to_string(dataPackage.channelId) + "/messages/" + std::to_string(dataPackage.messageId);
199 workload.callStack = "Messages::deleteMessageAsync()";
200 if (dataPackage.reason != "") {
201 workload.headersToInsert["X-Audit-Log-Reason"] = dataPackage.reason;
202 }
203 co_return Messages::httpsClient->submitWorkloadAndGetResult<void>(workload);
204 }
